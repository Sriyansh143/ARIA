   337→export default function MissionControlDashboard() {
   338→  // Single source of truth for the active tab is the global nav store,
   339→  // so any component (StatCard, Orion voice command, agent alerts, …) can
   340→  // navigate without prop-drilling. The shell subscribes here.
   341→  const tab = useNavStore((s) => s.tab as TabKey);
   342→  const navigateStore = useNavStore((s) => s.navigate);
   343→  const [sidebarOpen, setSidebarOpen] = useState(false);
   344→  const [paletteOpen, setPaletteOpen] = useState(false);
   345→  const [paletteKey, setPaletteKey] = useState(0);
   346→  const [globalSearchOpen, setGlobalSearchOpen] = useState(false);
   347→  const [notifOpen, setNotifOpen] = useState(false);
   348→  const [now, setNow] = useState<Date | null>(null);
   349→  const [theme, setTheme] = useState<'dark' | 'light'>('dark');
   350→  const [manageOpen, setManageOpen] = useState(false);
   351→  // Keyboard shortcuts overlay (press `?`).
   352→  const [shortcutsOpen, setShortcutsOpen] = useState(false);
   353→  // Tab personalization: hidden tabs + pinned tabs + custom order.
   354→  const [tabPrefs, setTabPrefs] = useState<{ hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }>({ hidden: [], pinned: [], order: [] });
   355→  // Orion voice-shell overlay mode (persisted). When true, a full-screen
   356→  // voice-first interface replaces the dashboard.
   357→  const [orionMode, setOrionMode] = useState(false);
   358→
   359→  const { data: metrics } = useApi<MetricsData>('/api/metrics', 8000);
   360→  const { data: commsData } = useApi<{ unread: number }>('/api/comms', 15000);
   361→  const commsUnread = commsData?.unread ?? 0;
   362→
   363→  // Branding — fetch once on mount (-1 = no polling). Drives the header
   364→  // app-name, version, footer powered-by, and chat-tab label.
   365→  const { data: brandingData } = useApi<{
   366→    config: {
   367→      appName: string;
   368→      version: string;
   369→      poweredBy?: string;
   370→      chatTabLabel?: string;
   371→      footerNote?: string;
   372→    };
   373→    defaults: { appName: string; version: string };
   374→  }>('/api/branding', -1);
   375→  const branding = brandingData?.config;
   376→  const appName = branding?.appName || 'JARVIS';
   377→  const appVersion = branding?.version || brandingData?.defaults.version || JARVIS.version;
   378→  const appPoweredBy = branding?.poweredBy || '';
   379→  const appCompany = branding?.company || 'Liafon Software Private Limited';
   380→  const chatTabLabel = branding?.chatTabLabel || 'JARVIS Chat';
   381→  const footerNote = branding?.footerNote || '';
   382→
   383→  // Resolve a tab's display label — chat label is driven by branding config.
   384→  const tabLabelOf = useCallback((t: TabDef) => (t.key === 'chat' ? chatTabLabel : t.label), [chatTabLabel]);
   385→
   386→  // Theme: load persisted preference on mount, default dark.
   387→  useEffect(() => {
   388→    const saved = (typeof window !== 'undefined' && localStorage.getItem('jarvis-theme')) as 'dark' | 'light' | null;
   389→    if (saved === 'light' || saved === 'dark') {
   390→      setTheme(saved);
   391→      document.documentElement.classList.toggle('light', saved === 'light');
   392→      document.documentElement.classList.toggle('dark', saved === 'dark');
   393→    }
   394→  }, []);
   395→
   396→  // Tab prefs: load persisted personalization on mount.
   397→  useEffect(() => {
   398→    try {
   399→      const raw = localStorage.getItem('jarvis-tab-prefs');
   400→      if (raw) {
   401→        const p = JSON.parse(raw);
   402→        setTabPrefs({ hidden: p.hidden ?? [], pinned: p.pinned ?? [], order: p.order ?? [] });
   403→      }
   404→    } catch { /* ignore */ }
   405→  }, []);
   406→
   407→  const updateTabPrefs = useCallback((updater: (prev: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => {
   408→    setTabPrefs((prev) => {
   409→      const next = updater(prev);
   410→      try { localStorage.setItem('jarvis-tab-prefs', JSON.stringify(next)); } catch { /* ignore */ }
   411→      return next;
   412→    });
   413→  }, []);
   414→
   415→  const toggleTheme = useCallback(() => {
   416→    setTheme((t) => {
   417→      const next = t === 'dark' ? 'light' : 'dark';
   418→      document.documentElement.classList.toggle('light', next === 'light');
   419→      document.documentElement.classList.toggle('dark', next === 'dark');
   420→      try { localStorage.setItem('jarvis-theme', next); } catch { /* ignore */ }
   421→      return next;
   422→    });
   423→  }, []);
   424→
   425→  // Live clock — only after mount to avoid hydration mismatch.
   426→  useEffect(() => {
   427→    setNow(new Date());
   428→    const id = setInterval(() => setNow(new Date()), 1000);
   429→    return () => clearInterval(id);
   430→  }, []);
   431→
   432→  const openPalette = useCallback(() => {
   433→    setPaletteKey((k) => k + 1);
   434→    setPaletteOpen(true);
   435→  }, []);
   436→
   437→  // Command palette shortcut (Cmd/Ctrl+K).
   438→  useEffect(() => {
   439→    const onKey = (e: KeyboardEvent) => {
   440→      if ((e.metaKey || e.ctrlKey) && e.key.toLowerCase() === 'k') {
   441→        e.preventDefault();
   442→        setPaletteOpen((o) => {
   443→          if (!o) setPaletteKey((k) => k + 1);
   444→          return !o;
   445→        });
   446→      }
   447→      if ((e.metaKey || e.ctrlKey) && e.shiftKey && e.key.toLowerCase() === 'f') {
   448→        e.preventDefault();
   449→        setGlobalSearchOpen((o) => !o);
   450→      }
   451→      // `?` (Shift+/) toggles the shortcuts overlay — but only when not typing in an input.
   452→      if (e.key === '?' && !isTypingTarget(e.target)) {
   453→        e.preventDefault();
   454→        setShortcutsOpen((o) => !o);
   455→      }
   456→      // `T` toggles theme (only when not typing + no modifiers).
   457→      if (e.key.toLowerCase() === 't' && !e.metaKey && !e.ctrlKey && !e.altKey && !isTypingTarget(e.target)) {
   458→        e.preventDefault();
   459→        toggleTheme();
   460→      }
   461→      if (e.key === 'Escape') {
   462→        setPaletteOpen(false);
   463→        setGlobalSearchOpen(false);
   464→        setNotifOpen(false);
   465→        setShortcutsOpen(false);
   466→      }
   467→    };
   468→    window.addEventListener('keydown', onKey);
   469→    return () => window.removeEventListener('keydown', onKey);
   470→  }, [toggleTheme]);
   471→
   472→  // Orion mode: default shell on app open. On first visit (no localStorage
   473→  // key), Orion opens automatically. After that, the user's last choice
   474→  // (open/closed) is persisted and restored.
   475→  useEffect(() => {
   476→    try {
   477→      const raw = localStorage.getItem('jarvis-orion-mode');
   478→      if (raw === null) {
   479→        // First visit — default to Orion open (hands-free voice shell).
   480→        setOrionMode(true);
   481→        localStorage.setItem('jarvis-orion-mode', '1');
   482→      } else if (raw === '1') {

   973→    refresh();

   975→  };
   976→

   978→  const [settings, setSettings] = useState<{ sound: boolean; desktop: boolean; mutedTypes: string[] }>({ sound: false, desktop: false, mutedTypes: [] });
   979→

   981→  useEffect(() => {
   982→    try {
   983→      const raw = localStorage.getItem('jarvis-notif-settings');

  1008→    acc[n.type] = (acc[n.type] ?? 0) + 1;
  1009→    return acc;
  1010→  }, {});
  1011→  const filteredVisible = filter === 'all' ? visibleNotifications : visibleNotifications.filter((n) => n.type === filter);
  1012→  const visibleUnread = visibleNotifications.filter((n) => !n.read).length;
  1013→  const visibleTotalBadge = visibleUnread + criticalFindings.length;
  1014→
  1015→  const markOne = async (id: string, read: boolean) => {
  1016→    await patchNotif(id, read);
  1017→    refresh();
  1018→  };
  1019→
  1020→  // Track previous unread count + seen notification IDs to detect new arrivals.
  1021→  const prevUnreadRef = useRef(0);
  1022→  const seenNotifIdsRef = useRef<Set<string>>(new Set());
  1023→  // Debounce timer for batching desktop notifications.
  1024→  const desktopBatchTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  1036→

  1038→      prevUnreadRef.current = currentUnread;
  1039→      return;
  1040→    }

  1047→        try {
  1048→          const ctx = new (window.AudioContext || (window as unknown as { webkitAudioContext: typeof AudioContext }).webkitAudioContext)();

  1104→              const notif = new Notification(title, {

  1107→                tag: 'jarvis-batch-' + Date.now(),
  1108→              });

  1111→                window.focus();
