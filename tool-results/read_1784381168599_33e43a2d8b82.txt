     1тЖТ     1тЖТ/**
     2тЖТ     2тЖТ * orion-intent.ts тАФ Orion Shell intent-routing engine.
     3тЖТ     3тЖТ *
     4тЖТ     4тЖТ * Pure-TS, zero-dependency, sync parser. No DB, no LLM. Runs in <1ms so the
     5тЖТ     5тЖТ * shell can show instant feedback (orb pulse + spoken ack) before the slow
     6тЖТ     6тЖТ * /api/orion/command endpoint does the real work.
     7тЖТ     7тЖТ *
     8тЖТ     8тЖТ * Taxonomy (14 intents):
     9тЖТ     9тЖТ *   navigate       тЖТ "show fleet", "open tasks", "go to payments"
    10тЖТ    10тЖТ *   create-task    тЖТ "create a task to...", "add task"
    11тЖТ    11тЖТ *   create-agent   тЖТ "spawn agent", "new agent for..."
    12тЖТ    12тЖТ *   run-skill      тЖТ "run skill web-search", "execute skill summarize"
    13тЖТ    13тЖТ *   send-comms     тЖТ "send message to orion", "tell agent", "notify"
    14тЖТ    14тЖТ *   health-check   тЖТ "health check", "system status"
    15тЖТ    15тЖТ *   sync-models    тЖТ "sync models", "update model list"
    16тЖТ    16тЖТ *   query-fleet    тЖТ "fleet status", "agent status"
    17тЖТ    17тЖТ *   query-revenue  тЖТ "revenue", "payments today"
    18тЖТ    18тЖТ *   query-tasks    тЖТ "task status", "what's pending"
    19тЖТ    19тЖТ *   set-theme      тЖТ "dark mode", "light mode"
    20тЖТ    20тЖТ *   search         тЖТ "search for..."
    21тЖТ    21тЖТ *   help           тЖТ "help", "what can you do"
    22тЖТ    22тЖТ *   chat (fallback)тЖТ everything else тАФ let the LLM handle
    23тЖТ    23тЖТ *
    24тЖТ    24тЖТ * Output shape:
    25тЖТ    25тЖТ *   { intent, tab?, action?, params?, confidence, response?, suggestions?, graph? }
    26тЖТ    26тЖТ *
    27тЖТ    27тЖТ * `response` is omitted for `chat` (the LLM writes it). For every other
    28тЖТ    28тЖТ * intent the parser returns a ready-to-speak string so the shell can ack
    29тЖТ    29тЖТ * instantly while the API round-trip is in flight.

   260тЖТ   260тЖТ/* ---- create-task ---- */
   261тЖТ   261тЖТfunction matchCreateTask(t: string): MatchResult | null {
   262тЖТ   262тЖТ  if (!/\b(create|add|new|make|schedule)\b/.test(t)) return null;
   263тЖТ   263тЖТ  if (!/\b(task|to-?do|ticket|job)\b/.test(t)) return null;
   264тЖТ   264тЖТ
   265тЖТ   265тЖТ  // Try to peel off a title: "create a task to ship the API" тЖТ "ship the API"
   266тЖТ   266тЖТ  // Also tolerates priority words between the article and the noun:
   267тЖТ   267тЖТ  //   "create a high priority task to review the API" тЖТ "review the API"
   268тЖТ   268тЖТ  //   "create a critical task to fix login"            тЖТ "fix login"
   269тЖТ   269тЖТ  let title = '';
   270тЖТ   270тЖТ  const m = t.match(
   271тЖТ   271тЖТ    /(?:create|add|new|make|schedule)\s+(?:a\s+)?(?:new\s+)?(?:(?:critical|urgent|high|medium|low|p[0-3])(?:\s+priority)?\s+)?(?:task|to-?do|ticket|job)\s+(?:to|for|called|named|titled|:)?\s*(.+)$/,
   272тЖТ   272тЖТ  );
   273тЖТ   273тЖТ  if (m) title = m[1].replace(/^["'`]|["'`]$/g, '').trim();
   274тЖТ   274тЖТ
   275тЖТ   275тЖТ  // Priority
   276тЖТ   276тЖТ  let priority = 'medium';
   277тЖТ   277тЖТ  if (/\b(critical|urgent|p0)\b/.test(t)) priority = 'critical';
   278тЖТ   278тЖТ  else if (/\bhigh\b|\bp1\b/.test(t)) priority = 'high';
   279тЖТ   279тЖТ  else if (/\blow\b|\bp3\b/.test(t)) priority = 'low';
   280тЖТ   280тЖТ
   281тЖТ   281тЖТ  const response = title
   282тЖТ   282тЖТ    ? `Creating a ${priority}-priority task: "${title}".`
   283тЖТ   283тЖТ    : `Creating a new ${priority}-priority task.`;
   284тЖТ   284тЖТ
   285тЖТ   285тЖТ  return {
   286тЖТ   286тЖТ    intent: 'create-task',
   287тЖТ   287тЖТ    confidence: 0.92,
   288тЖТ   288тЖТ    action: { type: 'create-task', title, priority },
   289тЖТ   289тЖТ    params: { title, priority },
   290тЖТ   290тЖТ    response,
   291тЖТ   291тЖТ    suggestions: ['Show pending tasks', 'Open tasks tab', 'Assign this to an agent'],
   292тЖТ   292тЖТ  };
   293тЖТ   293тЖТ}

   295тЖТ   295тЖТ/* ---- create-agent ---- */
   296тЖТ   296тЖТfunction matchCreateAgent(t: string): MatchResult | null {
   297тЖТ   297тЖТ  if (!/\b(spawn|create|new|make|build)\b/.test(t)) return null;
   298тЖТ   298тЖТ  if (!/\b(agent|sub-?agent|worker|bot)\b/.test(t)) return null;
   299тЖТ   299тЖТ
   300тЖТ   300тЖТ  // Parent agent: "spawn agent from orion", "spawn agent under vega"
   301тЖТ   301тЖТ  let parent = '';
   302тЖТ   302тЖТ  const pm = t.match(/\b(?:from|under|child of|offspring of)\s+([a-z][a-z0-9_-]{1,24})/);
   303тЖТ   303тЖТ  if (pm) parent = pm[1];
   304тЖТ   304тЖТ
   305тЖТ   305тЖТ  // Role: "for research", "to handle support"
   306тЖТ   306тЖТ  let role = '';
   307тЖТ   307тЖТ  const rm = t.match(/\b(?:for|to handle|to do|specializing in|specialized in)\s+(.+?)(?:\s+using|\s+with|$)/);
   308тЖТ   308тЖТ  if (rm) role = rm[1].replace(/\b(agent|sub agent|sub-agent)\b/g, '').trim();
   309тЖТ   309тЖТ
   310тЖТ   310тЖТ  // If no parent mentioned, default to "orion" (the orchestrator)
   311тЖТ   311тЖТ  if (!parent) parent = 'orion';
   312тЖТ   312тЖТ
   313тЖТ   313тЖТ  const response = role
   314тЖТ   314тЖТ    ? `Spawning a new sub-agent under ${parent} for ${role}.`
   315тЖТ   315тЖТ    : `Spawning a new sub-agent under ${parent}.`;
   316тЖТ   316тЖТ
   317тЖТ   317тЖТ  return {
   318тЖТ   318тЖТ    intent: 'create-agent',
   319тЖТ   319тЖТ    confidence: 0.9,
   320тЖТ   320тЖТ    action: { type: 'create-agent', parentCodename: parent, role },
   321тЖТ   321тЖТ    params: { parentCodename: parent, role },
   322тЖТ   322тЖТ    response,
   323тЖТ   323тЖТ    suggestions: ['Open spawned agents tab', 'Show fleet status', 'Send a message to the new agent'],
   324тЖТ   324тЖТ  };
   325тЖТ   325тЖТ}

   649тЖТ   649тЖТ/**
   650тЖТ   650тЖТ * Parse a free-text command into a structured intent.
   651тЖТ   651тЖТ *
   652тЖТ   652тЖТ * Always returns a ParsedIntent тАФ never throws. Falls back to `chat` with
   653тЖТ   653тЖТ * confidence 0 if no matcher fires, leaving the response to the LLM.
   654тЖТ   654тЖТ */
   655тЖТ   655тЖТexport function parseIntent(text: string): ParsedIntent {
   656тЖТ   656тЖТ  const raw = (text || '').trim();
   657тЖТ   657тЖТ  if (!raw) {
   658тЖТ   658тЖТ    return { intent: 'chat', confidence: 0 };
   659тЖТ   659тЖТ  }
   660тЖТ   660тЖТ  const t = raw.toLowerCase().replace(/[.,!?]+$/g, '').replace(/\s+/g, ' ');
   661тЖТ   661тЖТ
   662тЖТ   662тЖТ  for (const matcher of MATCHERS) {
   663тЖТ   663тЖТ    try {
   664тЖТ   664тЖТ      const r = matcher(t);
   665тЖТ   665тЖТ      if (r) {
   666тЖТ   666тЖТ        return {
   667тЖТ   667тЖТ          intent: r.intent,
   668тЖТ   668тЖТ          tab: r.tab,
   669тЖТ   669тЖТ          action: r.action,
   670тЖТ   670тЖТ          params: r.params,
   671тЖТ   671тЖТ          confidence: r.confidence,
   672тЖТ   672тЖТ          response: r.response,
   673тЖТ   673тЖТ          suggestions: r.suggestions,
   674тЖТ   674тЖТ          graph: r.graph,
   675тЖТ   675тЖТ        };
   676тЖТ   676тЖТ      }
   677тЖТ   677тЖТ    } catch {
   678тЖТ   678тЖТ      // a matcher bug should never break the shell тАФ keep going
   679тЖТ   679тЖТ    }
   680тЖТ   680тЖТ  }
   681тЖТ   681тЖТ
   682тЖТ   682тЖТ  // Fallback: let the LLM handle it.
   683тЖТ   683тЖТ  return { intent: 'chat', confidence: 0 };
   684тЖТ   684тЖТ}
