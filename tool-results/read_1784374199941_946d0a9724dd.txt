   275→export function timeAgo(date: Date | string | number): string {
   276→  const d = typeof date === 'object' ? date : new Date(date);
   277→  const s = Math.floor((Date.now() - d.getTime()) / 1000);
   278→  if (s < 5) return 'just now';
   279→  if (s < 60) return `${s}s ago`;
   280→  const m = Math.floor(s / 60);
   281→  if (m < 60) return `${m}m ago`;
   282→  const h = Math.floor(m / 60);
   283→  if (h < 24) return `${h}h ago`;
   284→  const dd = Math.floor(h / 24);
   285→  return `${dd}d ago`;
   286→}
   287→
   288→export function fmtTime(date: Date | string | number): string {
   289→  const d = typeof date === 'object' ? date : new Date(date);
   290→  return d.toLocaleTimeString('en-US', { hour12: false });
   291→}
