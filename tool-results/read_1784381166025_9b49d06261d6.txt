     1→/**
     2→ * orion-intent.ts — Orion Shell intent-routing engine.
     3→ *
     4→ * Pure-TS, zero-dependency, sync parser. No DB, no LLM. Runs in <1ms so the
     5→ * shell can show instant feedback (orb pulse + spoken ack) before the slow
     6→ * /api/orion/command endpoint does the real work.
     7→ *
     8→ * Taxonomy (14 intents):
     9→ *   navigate       → "show fleet", "open tasks", "go to payments"
    10→ *   create-task    → "create a task to...", "add task"
    11→ *   create-agent   → "spawn agent", "new agent for..."
    12→ *   run-skill      → "run skill web-search", "execute skill summarize"
    13→ *   send-comms     → "send message to orion", "tell agent", "notify"
    14→ *   health-check   → "health check", "system status"
    15→ *   sync-models    → "sync models", "update model list"
    16→ *   query-fleet    → "fleet status", "agent status"
    17→ *   query-revenue  → "revenue", "payments today"
    18→ *   query-tasks    → "task status", "what's pending"
    19→ *   set-theme      → "dark mode", "light mode"
    20→ *   search         → "search for..."
    21→ *   help           → "help", "what can you do"
    22→ *   chat (fallback)→ everything else — let the LLM handle
    23→ *
    24→ * Output shape:
    25→ *   { intent, tab?, action?, params?, confidence, response?, suggestions?, graph? }
    26→ *
    27→ * `response` is omitted for `chat` (the LLM writes it). For every other
    28→ * intent the parser returns a ready-to-speak string so the shell can ack
    29→ * instantly while the API round-trip is in flight.
    30→ */
    31→
    32→export type IntentName =
    33→  | 'navigate'
    34→  | 'create-task'
    35→  | 'create-agent'
    36→  | 'run-skill'
    37→  | 'send-comms'
    38→  | 'health-check'
    39→  | 'sync-models'
    40→  | 'query-fleet'
    41→  | 'query-revenue'
    42→  | 'query-tasks'
    43→  | 'set-theme'
    44→  | 'search'
    45→  | 'help'
    46→  | 'chat';
    47→
    48→export interface IntentGraph {
    49→  label: string;
    50→  value: number;
    51→}
    52→
    53→export interface IntentAction {
    54→  type: string;
    55→  [key: string]: unknown;
    56→}
    57→
    58→export interface ParsedIntent {
    59→  intent: IntentName;
    60→  tab?: string;
    61→  action?: IntentAction;
    62→  params?: Record<string, unknown>;
    63→  confidence: number; // 0..1
    64→  /** Spoken reply — omitted for `chat` (LLM writes it). */
    65→  response?: string;
    66→  suggestions?: string[];
    67→  graph?: IntentGraph[];
    68→}

   260→/* ---- create-task ---- */
   261→function matchCreateTask(t: string): MatchResult | null {
   262→  if (!/\b(create|add|new|make|schedule)\b/.test(t)) return null;
   263→  if (!/\b(task|to-?do|ticket|job)\b/.test(t)) return null;
   264→
   265→  // Try to peel off a title: "create a task to ship the API" → "ship the API"
   266→  // Also tolerates priority words between the article and the noun:
   267→  //   "create a high priority task to review the API" → "review the API"
   268→  //   "create a critical task to fix login"            → "fix login"
   269→  let title = '';
   270→  const m = t.match(
   271→    /(?:create|add|new|make|schedule)\s+(?:a\s+)?(?:new\s+)?(?:(?:critical|urgent|high|medium|low|p[0-3])(?:\s+priority)?\s+)?(?:task|to-?do|ticket|job)\s+(?:to|for|called|named|titled|:)?\s*(.+)$/,
   272→  );
   273→  if (m) title = m[1].replace(/^["'`]|["'`]$/g, '').trim();
   274→
   275→  // Priority
   276→  let priority = 'medium';
   277→  if (/\b(critical|urgent|p0)\b/.test(t)) priority = 'critical';
   278→  else if (/\bhigh\b|\bp1\b/.test(t)) priority = 'high';
   279→  else if (/\blow\b|\bp3\b/.test(t)) priority = 'low';
   280→
   281→  const response = title
   282→    ? `Creating a ${priority}-priority task: "${title}".`
   283→    : `Creating a new ${priority}-priority task.`;
   284→
   285→  return {
   286→    intent: 'create-task',
   287→    confidence: 0.92,
   288→    action: { type: 'create-task', title, priority },
   289→    params: { title, priority },
   290→    response,
   291→    suggestions: ['Show pending tasks', 'Open tasks tab', 'Assign this to an agent'],
   292→  };
   293→}

   295→/* ---- create-agent ---- */
   296→function matchCreateAgent(t: string): MatchResult | null {
   297→  if (!/\b(spawn|create|new|make|build)\b/.test(t)) return null;
   298→  if (!/\b(agent|sub-?agent|worker|bot)\b/.test(t)) return null;
   299→
   300→  // Parent agent: "spawn agent from orion", "spawn agent under vega"
   301→  let parent = '';
   302→  const pm = t.match(/\b(?:from|under|child of|offspring of)\s+([a-z][a-z0-9_-]{1,24})/);
   303→  if (pm) parent = pm[1];
   304→
   305→  // Role: "for research", "to handle support"
   306→  let role = '';
   307→  const rm = t.match(/\b(?:for|to handle|to do|specializing in|specialized in)\s+(.+?)(?:\s+using|\s+with|$)/);
   308→  if (rm) role = rm[1].replace(/\b(agent|sub agent|sub-agent)\b/g, '').trim();
   309→
   310→  // If no parent mentioned, default to "orion" (the orchestrator)
   311→  if (!parent) parent = 'orion';
   312→
   313→  const response = role
   314→    ? `Spawning a new sub-agent under ${parent} for ${role}.`
   315→    : `Spawning a new sub-agent under ${parent}.`;
   316→
   317→  return {
   318→    intent: 'create-agent',
   319→    confidence: 0.9,
   320→    action: { type: 'create-agent', parentCodename: parent, role },
   321→    params: { parentCodename: parent, role },
   322→    response,
   323→    suggestions: ['Open spawned agents tab', 'Show fleet status', 'Send a message to the new agent'],
   324→  };
   325→}

   327→/* ---- run-skill ---- */
   328→function matchRunSkill(t: string): MatchResult | null {
   329→  if (!/\b(run|execute|use|invoke|trigger)\b/.test(t)) return null;
   330→  if (!/\b(skill|web-?search|web-?reader|summarize|summarise|code-?gen|code-?review|forecast)\b/.test(t)) return null;
   331→
   332→  // Identify skill key
   333→  let skillKey = '';
   334→  if (/\bweb-?search\b/.test(t)) skillKey = 'web-search';
   335→  else if (/\bweb-?reader\b/.test(t)) skillKey = 'web-reader';
   336→  else if (/\b(summarize|summarise|summary)\b/.test(t)) skillKey = 'summarize';
   337→  else if (/\bcode-?gen\b/.test(t)) skillKey = 'code-gen';
   338→  else if (/\bcode-?review\b/.test(t)) skillKey = 'code-review';
   339→  else if (/\bforecast\b/.test(t)) skillKey = 'forecast';
   340→  else {
   341→    // generic: "run skill <key>"
   342→    const m = t.match(/skill\s+([a-z][a-z0-9_-]{1,32})/);
   343→    if (m) skillKey = m[1];
   344→  }
   345→  if (!skillKey) return null;
   346→
   347→  // Input: "run skill summarize on <text>", "use web-search for <query>"
   348→  let input = '';
   349→  const im = t.match(/(?:on|for|with|about|:)\s+(.+)$/);
   350→  if (im) input = im[1].replace(/^["'`]|["'`]$/g, '').trim();
   351→
   352→  const response = input
   353→    ? `Running the ${skillKey} skill on "${input.slice(0, 80)}".`
   354→    : `Running the ${skillKey} skill.`;
   355→
   356→  return {
   357→    intent: 'run-skill',
   358→    confidence: 0.9,
   359→    action: { type: 'run-skill', skillKey, input },
   360→    params: { skillKey, input },
   361→    response,
   362→    suggestions: ['Open skill runner', 'Show skill history', 'Run another skill'],
   363→  };
   364→}

   366→/* ---- send-comms ---- */
   367→function matchSendComms(t: string): MatchResult | null {
   368→  if (!/\b(send|tell|notify|message|ping|broadcast)\b/.test(t)) return null;
   369→  if (!/\b(message|agent|comms|notify|tell|ping|broadcast)\b/.test(t)) return null;
   370→
   371→  // recipient: "send message to orion" / "tell vega" / "notify all" / "broadcast"
   372→  let toAgent = 'BROADCAST';
   373→  let isBroadcast = /\b(broadcast|all agents|everyone|notify all)\b/.test(t);
   374→  const rm = t.match(/\b(?:to|tell|notify|ping)\s+([a-z][a-z0-9_-]{1,24})/);
   375→  if (rm && !isBroadcast) {
   376→    const cand = rm[1];
   377→    // filter out common stopwords that follow "tell/notify"
   378→    if (!['me', 'us', 'them', 'him', 'her', 'the', 'about', 'that', 'everyone', 'all'].includes(cand)) {
   379→      toAgent = cand;
   380→    } else if (cand === 'all' || cand === 'everyone') {
   381→      isBroadcast = true;
   382→    }
   383→  }
   384→
   385→  // subject + body
   386→  let body = '';
   387→  const bm = t.match(/(?:message|notify|tell|ping|broadcast)(?:\s+\w+){0,3}\s*[:\-]?\s*(.+)$/);
   388→  if (bm) body = bm[1].replace(/^["'`]|["'`]$/g, '').trim();
   389→  if (!body) {
   390→    const sm = t.match(/(?:saying|that|:)\s+(.+)$/);
   391→    if (sm) body = sm[1].trim();
   392→  }
   393→  if (!body) body = '(no body)';
   394→
   395→  const subject = body.length > 60 ? body.slice(0, 57) + '…' : body;
   396→  const response = isBroadcast
   397→    ? `Broadcasting: "${body.slice(0, 80)}".`
   398→    : `Sending a message to ${toAgent}: "${body.slice(0, 80)}".`;
   399→
   400→  return {
   401→    intent: 'send-comms',
   402→    confidence: 0.88,
   403→    action: { type: 'send-comms', toAgent, subject, body, priority: 'normal', thread: 'orion' },
   404→    params: { toAgent, subject, body, isBroadcast },
   405→    response,
   406→    suggestions: ['Open comms tab', 'Send another message', 'Show fleet status'],
   407→  };
   408→}

   452→/* ---- navigate ---- */
   453→function matchNavigate(t: string): MatchResult | null {
   454→  // "show fleet", "open tasks", "go to payments", "view health", "take me to models"
   455→  const m = t.match(/\b(?:show|open|go\s+to|view|visit|take\s+me\s+to|switch\s+to|navigate\s+to|jump\s+to|bring\s+up)\s+(?:the\s+|all\s+)?(.+)$/);
   456→  if (!m) return null;
   457→  const phrase = m[1].replace(/\b(tab|page|view|panel|section)\b/g, '').trim();
   458→  // try direct alias lookup
   459→  let tab = TAB_ALIASES[phrase];
   460→  // try last word (e.g. "open agent fleet" → "fleet")
   461→  if (!tab) {
   462→    const words = phrase.split(/\s+/);
   463→    for (let i = words.length - 1; i >= 0; i--) {
   464→      const w = words[i];
   465→      if (TAB_ALIASES[w]) { tab = TAB_ALIASES[w]; break; }
   466→    }
   467→  }
   468→  // try multi-word combinations
   469→  if (!tab) {
   470→    for (const k of Object.keys(TAB_ALIASES)) {
   471→      if (phrase.includes(k)) { tab = TAB_ALIASES[k]; break; }
   472→    }
   473→  }
   474→  if (!tab) return null;
   475→
   476→  const label = TAB_GROUP_LABEL[tab] ?? tab;
   477→  return {
   478→    intent: 'navigate',
   479→    confidence: 0.9,
   480→    tab,
   481→    action: { type: 'navigate', tab },
   482→    response: `Opening ${label}.`,
   483→    suggestions: [`Show ${label.toLowerCase()} status`, 'Take me to overview'],
   484→  };
   485→}

   487→/* ---- Bare single-word navigation: just "fleet" or "tasks" ---- */
   488→function matchBareNavigate(t: string): MatchResult | null {
   489→  const trimmed = t.trim().replace(/^the\s+/, '');
   490→  if (trimmed.length > 24) return null; // bare words only
   491→  const tab = TAB_ALIASES[trimmed];
   492→  if (!tab) return null;
   493→  const label = TAB_GROUP_LABEL[tab] ?? tab;
   494→  return {
   495→    intent: 'navigate',
   496→    confidence: 0.7,
   497→    tab,
   498→    action: { type: 'navigate', tab },
   499→    response: `Opening ${label}.`,

   505→function matchMakePlan(t: string): MatchResult | null {

   507→  const m = t.match(/^(?:plan|make\s+a\s+plan|create\s+a\s+plan|decompose|break\s+down|analyze\s+and\s+plan)\s*[:\s]*(.+)$/);
   508→  if (m) {

   516→      suggestions: ['Execute plan', 'Modify plan', 'Save as tasks'],
   517→    };
   518→  }
   519→  // "plan to...", "plan for..."
   520→  const m2 = t.match(/^plan\s+(?:to|for)\s+(.+)$/);
   521→  if (m2) {
   522→    return {
   523→      intent: 'make-plan',
   524→      confidence: 0.92,

   532→}

   604→/* ---- browse (browser automation) ---- */
   605→function matchBrowse(t: string): MatchResult | null {
   606→  // "browse to github.com", "open website cnn.com", "go to news.ycombinator.com"
   607→  const m = t.match(/^(?:browse\s+to|open\s+website|go\s+to\s+website|visit\s+site|browse)\s+(.+)$/);
   608→  if (m) {
   609→    let url = m[1].trim();
   610→    if (!/^https?:\/\//i.test(url)) url = `https://${url}`;
   611→    return {
   612→      intent: 'browse',
   613→      confidence: 0.92,
   614→      action: { type: 'browse', url },
   615→      params: { url },
   616→      response: `Opening ${url} in the browser.`,
   617→      suggestions: ['Take a screenshot', 'Extract page content', 'Click an element'],
   618→    };
   619→  }
   620→  return null;
   621→}

   649→/**
   650→ * Parse a free-text command into a structured intent.
   651→ *
   652→ * Always returns a ParsedIntent — never throws. Falls back to `chat` with
   653→ * confidence 0 if no matcher fires, leaving the response to the LLM.
   654→ */
   655→export function parseIntent(text: string): ParsedIntent {
   656→  const raw = (text || '').trim();
   657→  if (!raw) {
   658→    return { intent: 'chat', confidence: 0 };
   659→  }
   660→  const t = raw.toLowerCase().replace(/[.,!?]+$/g, '').replace(/\s+/g, ' ');
   661→
   662→  for (const matcher of MATCHERS) {
   663→    try {
   664→      const r = matcher(t);
   665→      if (r) {
   666→        return {
   667→          intent: r.intent,
   668→          tab: r.tab,
   669→          action: r.action,
   670→          params: r.params,
   671→          confidence: r.confidence,
   672→          response: r.response,
   673→          suggestions: r.suggestions,
   674→          graph: r.graph,
   675→        };
   676→      }
   677→    } catch {
   678→      // a matcher bug should never break the shell — keep going
   679→    }
   680→  }
   681→
   682→  // Fallback: let the LLM handle it.
   683→  return { intent: 'chat', confidence: 0 };
   684→}

   878→
   879→export function detectContext(text: string): string | null {
   880→  const t = (text || '').toLowerCase();
   881→  if (!t) return null;
   882→  for (const { ctx, words } of CONTEXT_KEYWORDS) {
   883→    if (words.some((w) => t.includes(w))) return ctx;
   884→  }
   885→  return null;
   886→}
