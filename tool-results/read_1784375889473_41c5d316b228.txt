   333→export default function MissionControlDashboard() {
   334→  // Single source of truth for the active tab is the global nav store,
   335→  // so any component (StatCard, Orion voice command, agent alerts, …) can
   336→  // navigate without prop-drilling. The shell subscribes here.
   337→  const tab = useNavStore((s) => s.tab as TabKey);
   338→  const navigateStore = useNavStore((s) => s.navigate);
   339→  const [sidebarOpen, setSidebarOpen] = useState(false);
   340→  const [paletteOpen, setPaletteOpen] = useState(false);
   341→  const [paletteKey, setPaletteKey] = useState(0);
   342→  const [globalSearchOpen, setGlobalSearchOpen] = useState(false);
   343→  const [notifOpen, setNotifOpen] = useState(false);
   344→  const [now, setNow] = useState<Date | null>(null);
   345→  const [theme, setTheme] = useState<'dark' | 'light'>('dark');
   346→  const [manageOpen, setManageOpen] = useState(false);
   347→  // Keyboard shortcuts overlay (press `?`).
   348→  const [shortcutsOpen, setShortcutsOpen] = useState(false);
   349→  // Tab personalization: hidden tabs + pinned tabs + custom order.
   350→  const [tabPrefs, setTabPrefs] = useState<{ hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }>({ hidden: [], pinned: [], order: [] });
   351→  // Orion voice-shell overlay mode (persisted). When true, a full-screen
   352→  // voice-first interface replaces the dashboard.
   353→  const [orionMode, setOrionMode] = useState(false);
   354→
   355→  const { data: metrics } = useApi<MetricsData>('/api/metrics', 8000);
   356→  const { data: commsData } = useApi<{ unread: number }>('/api/comms', 15000);
   357→  const commsUnread = commsData?.unread ?? 0;
   358→
   359→  // Branding — fetch once on mount (-1 = no polling). Drives the header
   360→  // app-name, version, footer powered-by, and chat-tab label.
   361→  const { data: brandingData } = useApi<{
   362→    config: {
   363→      appName: string;
   364→      version: string;
   365→      poweredBy?: string;
   366→      chatTabLabel?: string;
   367→      footerNote?: string;
   368→    };
   369→    defaults: { appName: string; version: string };
   370→  }>('/api/branding', -1);
   371→  const branding = brandingData?.config;
   372→  const appName = branding?.appName || 'JARVIS';
   373→  const appVersion = branding?.version || brandingData?.defaults.version || JARVIS.version;
   374→  const appPoweredBy = branding?.poweredBy || '';
   375→  const appCompany = branding?.company || 'Liafon Software Private Limited';
   376→  const chatTabLabel = branding?.chatTabLabel || 'JARVIS Chat';
   377→  const footerNote = branding?.footerNote || '';
   378→
   379→  // Resolve a tab's display label — chat label is driven by branding config.
   380→  const tabLabelOf = useCallback((t: TabDef) => (t.key === 'chat' ? chatTabLabel : t.label), [chatTabLabel]);
   381→
   382→  // Theme: load persisted preference on mount, default dark.
   383→  useEffect(() => {
   384→    const saved = (typeof window !== 'undefined' && localStorage.getItem('jarvis-theme')) as 'dark' | 'light' | null;
   385→    if (saved === 'light' || saved === 'dark') {
   386→      setTheme(saved);
   387→      document.documentElement.classList.toggle('light', saved === 'light');
   388→      document.documentElement.classList.toggle('dark', saved === 'dark');
   389→    }
   390→  }, []);
   391→
   392→  // Tab prefs: load persisted personalization on mount.
   393→  useEffect(() => {
   394→    try {
   395→      const raw = localStorage.getItem('jarvis-tab-prefs');
   396→      if (raw) {
   397→        const p = JSON.parse(raw);
   398→        setTabPrefs({ hidden: p.hidden ?? [], pinned: p.pinned ?? [], order: p.order ?? [] });
   399→      }
   400→    } catch { /* ignore */ }
   401→  }, []);
   402→
   403→  const updateTabPrefs = useCallback((updater: (prev: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => {
   404→    setTabPrefs((prev) => {
   405→      const next = updater(prev);
   406→      try { localStorage.setItem('jarvis-tab-prefs', JSON.stringify(next)); } catch { /* ignore */ }
   407→      return next;
   408→    });
   409→  }, []);
   410→
   411→  const toggleTheme = useCallback(() => {
   412→    setTheme((t) => {
   413→      const next = t === 'dark' ? 'light' : 'dark';
   414→      document.documentElement.classList.toggle('light', next === 'light');
   415→      document.documentElement.classList.toggle('dark', next === 'dark');
   416→      try { localStorage.setItem('jarvis-theme', next); } catch { /* ignore */ }
   417→      return next;
   418→    });
   419→  }, []);
   420→
   421→  // Live clock — only after mount to avoid hydration mismatch.
   422→  useEffect(() => {
   423→    setNow(new Date());
   424→    const id = setInterval(() => setNow(new Date()), 1000);
   425→    return () => clearInterval(id);
   426→  }, []);
   427→
   428→  const openPalette = useCallback(() => {
   429→    setPaletteKey((k) => k + 1);
   430→    setPaletteOpen(true);
   431→  }, []);
   432→
   433→  // Command palette shortcut (Cmd/Ctrl+K).
   434→  useEffect(() => {
   435→    const onKey = (e: KeyboardEvent) => {
   436→      if ((e.metaKey || e.ctrlKey) && e.key.toLowerCase() === 'k') {
   437→        e.preventDefault();
   438→        setPaletteOpen((o) => {
   439→          if (!o) setPaletteKey((k) => k + 1);
   440→          return !o;
   441→        });
   442→      }
   443→      if ((e.metaKey || e.ctrlKey) && e.shiftKey && e.key.toLowerCase() === 'f') {
   444→        e.preventDefault();
   445→        setGlobalSearchOpen((o) => !o);
   446→      }
   447→      // `?` (Shift+/) toggles the shortcuts overlay — but only when not typing in an input.
   448→      if (e.key === '?' && !isTypingTarget(e.target)) {
   449→        e.preventDefault();
   450→        setShortcutsOpen((o) => !o);
   451→      }
   452→      // `T` toggles theme (only when not typing + no modifiers).
   453→      if (e.key.toLowerCase() === 't' && !e.metaKey && !e.ctrlKey && !e.altKey && !isTypingTarget(e.target)) {
   454→        e.preventDefault();
   455→        toggleTheme();
   456→      }
   457→      if (e.key === 'Escape') {
   458→        setPaletteOpen(false);
   459→        setGlobalSearchOpen(false);
   460→        setNotifOpen(false);
   461→        setShortcutsOpen(false);
   462→      }
   463→    };
   464→    window.addEventListener('keydown', onKey);
   465→    return () => window.removeEventListener('keydown', onKey);
   466→  }, [toggleTheme]);
   467→
   468→  // Orion mode: load persisted preference on mount, then persist on change.
   469→  useEffect(() => {
   470→    try {
   471→      const raw = localStorage.getItem('jarvis-orion-mode');
   472→      if (raw === '1') setOrionMode(true);
   473→    } catch { /* ignore */ }
   474→  }, []);
   475→  useEffect(() => {
   476→    try { localStorage.setItem('jarvis-orion-mode', orionMode ? '1' : '0'); } catch { /* ignore */ }
   477→  }, [orionMode]);
   478→
   479→  // Orion mode shortcut (Cmd/Ctrl+Shift+O).
   480→  useEffect(() => {
   481→    const onKey = (e: KeyboardEvent) => {

   843→/**

   896→function activityColor(typeOrLevel: string): string {
   897→  switch (typeOrLevel) {
   898→    case 'error': return JARVIS.colors.red;
   899→    case 'success': return JARVIS.colors.green;
   900→    case 'warn': return JARVIS.colors.amber;
   901→    case 'warning': return JARVIS.colors.amber;
