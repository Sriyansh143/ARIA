   174→  const [manageOpen, setManageOpen] = useState(false);
   175→  // Tab personalization: hidden tabs + pinned tabs + custom order.
   176→  const [tabPrefs, setTabPrefs] = useState<{ hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }>({ hidden: [], pinned: [], order: [] });
   177→  // Orion voice-shell overlay mode (persisted). When true, a full-screen
   178→  // voice-first interface replaces the dashboard.

   186→  // app-name, version, footer powered-by, and chat-tab label.

   212→      setTheme(saved);
   213→      document.documentElement.classList.toggle('light', saved === 'light');

   219→  useEffect(() => {

   221→      const raw = localStorage.getItem('jarvis-tab-prefs');

   223→        const p = JSON.parse(raw);
   224→        setTabPrefs({ hidden: p.hidden ?? [], pinned: p.pinned ?? [], order: p.order ?? [] });
   225→      }
   226→    } catch { /* ignore */ }
   227→  }, []);

   230→    setTabPrefs((prev) => {
   231→      const next = updater(prev);
   232→      try { localStorage.setItem('jarvis-tab-prefs', JSON.stringify(next)); } catch { /* ignore */ }
   233→      return next;
   234→    });
   235→  }, []);

   238→    setTheme((t) => {
   239→      const next = t === 'dark' ? 'light' : 'dark';
   240→      document.documentElement.classList.toggle('light', next === 'light');
   241→      document.documentElement.classList.toggle('dark', next === 'dark');
   242→      try { localStorage.setItem('jarvis-theme', next); } catch { /* ignore */ }
   243→      return next;
   244→    });
   245→  }, []);

   247→  // Live clock — only after mount to avoid hydration mismatch.

   249→    setNow(new Date());

   251→    return () => clearInterval(id);
   252→  }, []);
   253→
   254→  const openPalette = useCallback(() => {
   255→    setPaletteKey((k) => k + 1);
   256→    setPaletteOpen(true);

   262→      if ((e.metaKey || e.ctrlKey) && e.key.toLowerCase() === 'k') {
   263→        e.preventDefault();
   264→        setPaletteOpen((o) => {
   265→          if (!o) setPaletteKey((k) => k + 1);
   266→          return !o;
   267→        });
   268→      }
   269→      if ((e.metaKey || e.ctrlKey) && e.shiftKey && e.key.toLowerCase() === 'f') {
   270→        e.preventDefault();
   271→        setGlobalSearchOpen((o) => !o);
   272→      }
   273→      if (e.key === 'Escape') {
   274→        setPaletteOpen(false);
   275→        setGlobalSearchOpen(false);
   276→        setNotifOpen(false);
   277→      }
   278→    };
   279→    window.addEventListener('keydown', onKey);
   280→    return () => window.removeEventListener('keydown', onKey);
   281→  }, []);

   299→        setOrionMode((o) => !o);
   300→      }
   301→    };

   303→    return () => window.removeEventListener('keydown', onKey);

   324→  const cur = TABS.find((t) => t.key === tab)!;

   347→            </div>

   349→              <div className="text-sm font-bold tracking-tight jarvis-text-gradient">{appName}</div>

   671→/* ---------- Command palette ---------- */
   672→function CommandPalette({ open, onClose, onNavigate }: { open: boolean; onClose: () => void; onNavigate: (t: TabKey) => void }) {
   673→  // Fresh mount (via key in parent) ensures q/sel start empty each open.
   674→  const [q, setQ] = useState('');
   675→  const [sel, setSel] = useState(0);
   676→
   677→  const results = useMemo(() => {
   678→    const ql = q.toLowerCase();
   679→    return TABS.filter((t) => !ql || t.label.toLowerCase().includes(ql) || t.group.toLowerCase().includes(ql));
   680→  }, [q]);
   681→
   682→  useEffect(() => {
   683→    if (!open) return;
   684→    const onKey = (e: KeyboardEvent) => {
   685→      if (e.key === 'ArrowDown') { e.preventDefault(); setSel((s) => Math.min(s + 1, results.length - 1)); }
   686→      if (e.key === 'ArrowUp') { e.preventDefault(); setSel((s) => Math.max(s - 1, 0)); }
   687→      if (e.key === 'Enter') { e.preventDefault(); const r = results[sel]; if (r) onNavigate(r.key); }
   688→    };
   689→    window.addEventListener('keydown', onKey);
   690→    return () => window.removeEventListener('keydown', onKey);
   691→  }, [open, results, sel, onNavigate]);
   692→
   693→  return (
   694→    <AnimatePresence>
   695→      {open && (
   696→        <motion.div
   697→          className="fixed inset-0 z-50 flex items-start justify-center pt-[12vh] px-4"
   698→          initial={{ opacity: 0 }}
   699→          animate={{ opacity: 1 }}
   700→          exit={{ opacity: 0 }}
   701→        >
   702→          <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   703→          <motion.div
   704→            initial={{ opacity: 0, scale: 0.96, y: -8 }}
   705→            animate={{ opacity: 1, scale: 1, y: 0 }}
   706→            exit={{ opacity: 0, scale: 0.96, y: -8 }}
   707→            transition={{ duration: 0.18 }}
   708→            className="relative w-full max-w-xl jarvis-panel p-0 overflow-hidden"
   709→          >
   710→            <div className="flex items-center gap-2.5 px-4 h-12 border-b border-[var(--j-border)]">
   711→              <Search className="h-4 w-4 text-[var(--j-text-mute)]" />
   712→              <input
   713→                autoFocus
   714→                value={q}
   715→                onChange={(e) => { setQ(e.target.value); setSel(0); }}
   716→                placeholder="Search tabs and actions…"
   717→                className="flex-1 bg-transparent outline-none text-sm text-[var(--j-text)] placeholder:text-[var(--j-text-mute)]"
   718→              />
   719→              <kbd className="jarvis-mono text-[9px] px-1.5 py-0.5 rounded border border-[var(--j-border)] text-[var(--j-text-mute)]">ESC</kbd>
   720→            </div>
   721→            <div className="max-h-72 overflow-y-auto jarvis-scroll p-2">
   722→              {results.length ? results.map((t, i) => {
   723→                const Icon = t.icon;
   724→                const active = i === sel;
   725→                return (
   726→                  <button
   727→                    key={t.key}
   728→                    onMouseEnter={() => setSel(i)}
   729→                    onClick={() => onNavigate(t.key)}
   730→                    className={cn('w-full flex items-center gap-3 px-3 py-2 rounded-md text-sm transition-colors', active ? 'bg-[var(--j-panel-soft)] text-[var(--j-text)]' : 'text-[var(--j-text-dim)]')}
   731→                  >
   732→                    <Icon className="h-4 w-4" style={{ color: active ? t.accent : undefined }} />
   733→                    <span className="flex-1 text-left">{t.label}</span>
   734→                    <span className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">{t.group}</span>
   735→                  </button>
   736→                );
   737→              }) : (
   738→                <div className="px-3 py-8 text-center text-xs text-[var(--j-text-mute)]">No results</div>
   739→              )}
   740→            </div>
   741→          </motion.div>
   742→        </motion.div>
   743→      )}
   744→    </AnimatePresence>
   745→  );
   746→}

   748→/* ---------- Manage Tabs modal (personalization) ---------- */
   749→function ManageTabsModal({
   750→  onClose,
   751→  tabPrefs,
   752→  updateTabPrefs,
   753→  onNavigate,
   754→}: {
   755→  onClose: () => void;
   756→  tabPrefs: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] };
   757→  updateTabPrefs: (updater: (prev: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => void;
   758→  onNavigate: (t: TabKey) => void;
   759→}) {
   760→  const toggleHidden = (key: TabKey) => {
   761→    updateTabPrefs((prev) => ({
   762→      ...prev,
   763→      hidden: prev.hidden.includes(key) ? prev.hidden.filter((k) => k !== key) : [...prev.hidden, key],
   764→    }));
   765→  };
   766→  const togglePinned = (key: TabKey) => {
   767→    updateTabPrefs((prev) => ({
   768→      ...prev,
   769→      pinned: prev.pinned.includes(key) ? prev.pinned.filter((k) => k !== key) : [...prev.pinned, key],
   770→    }));
   771→  };
   772→  const resetAll = () => {
   773→    updateTabPrefs(() => ({ hidden: [], pinned: [], order: [] }));
   774→  };
   775→  const visibleCount = TABS.length - tabPrefs.hidden.length;
   776→
   777→  return (
   778→    <motion.div className="fixed inset-0 z-50 flex items-center justify-center p-4" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }}>
   779→      <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   780→      <motion.div initial={{ scale: 0.96, opacity: 0 }} animate={{ scale: 1, opacity: 1 }} exit={{ scale: 0.96, opacity: 0 }} className="relative w-full max-w-2xl jarvis-panel p-0 overflow-hidden max-h-[85vh] flex flex-col">
   781→        <div className="flex items-center justify-between px-5 py-3 border-b border-[var(--j-border)]">
   782→          <div className="flex items-center gap-2">
   783→            <Sliders className="h-4 w-4 text-[var(--j-cyan)]" />
   784→            <h3 className="jarvis-mono text-sm uppercase text-[var(--j-cyan)]">Manage Tabs</h3>
   785→            <span className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">{visibleCount} visible · {tabPrefs.pinned.length} pinned · {tabPrefs.hidden.length} hidden</span>
   786→          </div>
   787→          <div className="flex items-center gap-2">
   788→            <button onClick={resetAll} className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)] hover:text-[var(--j-cyan)]">reset</button>
   789→            <button onClick={onClose} className="text-[var(--j-text-mute)] hover:text-[var(--j-text)]"><X className="h-4 w-4" /></button>
   790→          </div>
   791→        </div>
   792→        <div className="overflow-y-auto jarvis-scroll p-3">
   793→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-2">
   794→            {TABS.map((t) => {
   795→              const isHidden = tabPrefs.hidden.includes(t.key);
   796→              const isPinned = tabPrefs.pinned.includes(t.key);
   797→              const Icon = t.icon;
   798→              return (
   799→                <div
   800→                  key={t.key}
   801→                  className={`flex items-center gap-2.5 p-2.5 rounded-lg border transition-all ${isHidden ? 'border-[var(--j-border-soft)] opacity-50' : 'border-[var(--j-border)] bg-[var(--j-panel-soft)]/40'}`}
   802→                >
   803→                  <button
   804→                    onClick={() => togglePinned(t.key)}
   805→                    className={`flex h-7 w-7 items-center justify-center rounded-md shrink-0 transition-colors ${isPinned ? 'text-[var(--j-violet)]' : 'text-[var(--j-text-mute)] hover:text-[var(--j-violet)]'}`}
   806→                    title={isPinned ? 'Unpin' : 'Pin to top'}
   807→                  >
   808→                    <Pin className={`h-3.5 w-3.5 ${isPinned ? 'fill-[var(--j-violet)]' : ''}`} />
   809→                  </button>
   810→                  <div className="flex h-7 w-7 items-center justify-center rounded-md shrink-0" style={{ background: `${t.accent}1a`, border: `1px solid ${t.accent}33`, color: t.accent }}>
   811→                    <Icon className="h-3.5 w-3.5" />
   812→                  </div>
   813→                  <button onClick={() => !isHidden && onNavigate(t.key)} className="flex-1 text-left min-w-0">
   814→                    <div className="text-xs text-[var(--j-text)] truncate">{t.label}</div>
   815→                    <div className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">{isPinned ? 'pinned · ' : ''}{t.group}</div>
   816→                  </button>
   817→                  <button
   818→                    onClick={() => toggleHidden(t.key)}
   819→                    className={`flex h-7 w-7 items-center justify-center rounded-md shrink-0 transition-colors ${isHidden ? 'text-[var(--j-text-mute)] hover:text-[var(--j-green)]' : 'text-[var(--j-green)] hover:text-[var(--j-red)]'}`}
   820→                    title={isHidden ? 'Show' : 'Hide'}
   821→                  >
   822→                    {isHidden ? <EyeOff className="h-3.5 w-3.5" /> : <Eye className="h-3.5 w-3.5" />}
   823→                  </button>
   824→                </div>
   825→              );
   826→            })}
   827→          </div>
   828→        </div>
   829→        <div className="px-5 py-2.5 border-t border-[var(--j-border)] bg-[var(--j-panel-soft)]/40 flex items-center justify-between">
   830→          <span className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">preferences saved to this browser</span>
   831→          <button onClick={onClose} className="jarvis-mono text-[10px] uppercase px-3 py-1 rounded jarvis-btn-accent border-0">Done</button>
   832→        </div>
   833→      </motion.div>

   838→/* ---------- Global search overlay (Cmd+Shift+F) ---------- */
   839→interface SearchResult {
   840→  id: string;
   841→  type: 'agent' | 'task' | 'memory' | 'comms' | 'skill';
   842→  title: string;
   843→  subtitle: string;
   844→  meta: string;
   845→  color: string;
   846→  href: TabKey;
   847→  score: number;
   848→}
   849→
   850→function GlobalSearch({ open, onClose, onNavigate }: { open: boolean; onClose: () => void; onNavigate: (t: TabKey) => void }) {
   851→  const [q, setQ] = useState('');
   852→  const [results, setResults] = useState<SearchResult[]>([]);
   853→  const [loading, setLoading] = useState(false);
   854→  const [sel, setSel] = useState(0);
   855→  const [savedSearches, setSavedSearches] = useState<string[]>([]);
   856→
   857→  // Load saved searches from localStorage on mount.
   858→  useEffect(() => {
   859→    try {
   860→      const raw = localStorage.getItem('jarvis-saved-searches');
   861→      if (raw) setSavedSearches(JSON.parse(raw));
   862→    } catch { /* ignore */ }
   863→  }, []);
   864→
   865→  const saveSearch = (term: string) => {
   866→    const t = term.trim();
   867→    if (!t) return;
   868→    setSavedSearches((prev) => {
   869→      const next = prev.includes(t) ? prev.filter((s) => s !== t) : [...prev, t].slice(-8);
   870→      try { localStorage.setItem('jarvis-saved-searches', JSON.stringify(next)); } catch { /* ignore */ }
   871→      return next;
   872→    });
   873→  };
   874→
   875→  // Debounced search.
   876→  useEffect(() => {
   877→    if (!open) { setQ(''); setResults([]); setSel(0); return; }
   878→    if (!q.trim()) { setResults([]); return; }
   879→    setLoading(true);
   880→    const id = setTimeout(async () => {
   881→      try {
   882→        const res = await fetch(`/api/search?q=${encodeURIComponent(q)}`, { cache: 'no-store' });
   883→        const json = await res.json();
   884→        setResults(json.results ?? []);
   885→        setSel(0);
   886→      } catch { /* ignore */ } finally {
   887→        setLoading(false);
   888→      }
   889→    }, 200);
   890→    return () => clearTimeout(id);
   891→  }, [q, open]);
   892→
   893→  useEffect(() => {
   894→    if (!open) return;
   895→    const onKey = (e: KeyboardEvent) => {
   896→      if (e.key === 'ArrowDown') { e.preventDefault(); setSel((s) => Math.min(s + 1, results.length - 1)); }
   897→      if (e.key === 'ArrowUp') { e.preventDefault(); setSel((s) => Math.max(s - 1, 0)); }
   898→      if (e.key === 'Enter') { e.preventDefault(); const r = results[sel]; if (r) { onNavigate(r.href); onClose(); } }
   899→    };
   900→    window.addEventListener('keydown', onKey);
   901→    return () => window.removeEventListener('keydown', onKey);
   902→  }, [open, results, sel, onNavigate, onClose]);
   903→
   904→  const typeIcon: Record<string, typeof Bot> = { agent: Bot, task: ListTodo, memory: Database, comms: MessagesSquare, skill: Sparkles };
   905→
   906→  return (
   907→    <AnimatePresence>
   908→      {open && (
   909→        <motion.div className="fixed inset-0 z-50 flex items-start justify-center pt-[10vh] px-4" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }}>
   910→          <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   911→          <motion.div initial={{ scale: 0.96, opacity: 0, y: -8 }} animate={{ scale: 1, opacity: 1, y: 0 }} exit={{ scale: 0.96, opacity: 0, y: -8 }} className="relative w-full max-w-2xl jarvis-panel p-0 overflow-hidden">
   912→            <div className="flex items-center gap-2.5 px-4 h-12 border-b border-[var(--j-border)]">
   913→              <Search className="h-4 w-4 text-[var(--j-cyan)]" />
   914→              <input
   915→                autoFocus
   916→                value={q}
   917→                onChange={(e) => setQ(e.target.value)}
   918→                placeholder="Search agents, tasks, memory, comms, skills…"
   919→                className="flex-1 bg-transparent outline-none text-sm text-[var(--j-text)] placeholder:text-[var(--j-text-mute)]"
   920→              />
   921→              {loading && <Loader2 className="h-3.5 w-3.5 animate-spin text-[var(--j-cyan)]" />}
   922→              {q.trim() && (
   923→                <button onClick={() => saveSearch(q)} className="flex items-center gap-1 jarvis-mono text-[9px] uppercase text-[var(--j-cyan)] hover:underline shrink-0">
   924→                  <Star className="h-3 w-3" /> {savedSearches.includes(q.trim()) ? 'saved' : 'save'}
   925→                </button>
   926→              )}
   927→              <kbd className="jarvis-mono text-[9px] px-1.5 py-0.5 rounded border border-[var(--j-border)] text-[var(--j-text-mute)]">ESC</kbd>
   928→            </div>
   929→            <div className="max-h-96 overflow-y-auto jarvis-scroll p-2">
   930→              {results.length > 0 ? (
   931→                results.map((r, i) => {
   932→                  const Icon = typeIcon[r.type] ?? Search;
   933→                  const active = i === sel;
   934→                  return (
   935→                    <button
   936→                      key={r.id}
   937→                      onMouseEnter={() => setSel(i)}
   938→                      onClick={() => { onNavigate(r.href); onClose(); }}
   939→                      className={cn('w-full flex items-center gap-3 px-3 py-2.5 rounded-md text-left transition-colors', active ? 'bg-[var(--j-panel-soft)]' : 'hover:bg-[var(--j-panel-soft)]/60')}
   940→                    >
   941→                      <div className="flex h-8 w-8 items-center justify-center rounded-md shrink-0" style={{ background: `${r.color}1a`, border: `1px solid ${r.color}33`, color: r.color }}>
   942→                        <Icon className="h-4 w-4" />
   943→                      </div>
   944→                      <div className="min-w-0 flex-1">
   945→                        <div className="text-sm text-[var(--j-text)] truncate">{r.title}</div>
   946→                        <div className="text-[11px] text-[var(--j-text-dim)] truncate">{r.subtitle}</div>
   947→                      </div>
   948→                      <div className="text-right shrink-0">
   949→                        <div className="jarvis-mono text-[9px] uppercase px-1.5 py-0.5 rounded" style={{ color: r.color, background: `${r.color}1a` }}>{r.type}</div>
   950→                        <div className="jarvis-mono text-[9px] text-[var(--j-text-mute)] mt-0.5">{r.meta}</div>
   951→                      </div>
   952→                    </button>
   953→                  );
   954→                })
   955→              ) : q.trim() ? (
   956→                <div className="px-3 py-10 text-center">
   957→                  <Search className="h-8 w-8 mx-auto mb-2 text-[var(--j-text-mute)] opacity-40" />
   958→                  <div className="text-sm text-[var(--j-text-mute)]">No results for "{q}"</div>
   959→                </div>
   960→              ) : savedSearches.length > 0 ? (
   961→                <div className="px-3 py-4">
   962→                  <div className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)] mb-2 flex items-center gap-1.5"><Star className="h-3 w-3" /> saved searches</div>
   963→                  <div className="flex flex-wrap gap-2">
   964→                    {savedSearches.map((s) => (
   965→                      <button key={s} onClick={() => setQ(s)} className="group flex items-center gap-1.5 text-xs px-2.5 py-1.5 rounded-md border border-[var(--j-border)] bg-[var(--j-panel-soft)] text-[var(--j-text-dim)] hover:border-[var(--j-cyan)] hover:text-[var(--j-cyan)] transition-colors">
   966→                        <Search className="h-3 w-3" /> {s}
   967→                        <span onClick={(e) => { e.stopPropagation(); saveSearch(s); }} className="text-[var(--j-text-mute)] hover:text-[var(--j-red)] ml-0.5"><X className="h-2.5 w-2.5" /></span>
   968→                      </button>
   969→                    ))}
   970→                  </div>
   971→                </div>
   972→              ) : (
   973→                <div className="px-3 py-10 text-center">
   974→                  <div className="text-sm text-[var(--j-text-mute)]">Type to search across the fleet</div>
   975→                  <div className="jarvis-mono text-[10px] text-[var(--j-text-mute)] mt-1">agents · tasks · memory · comms · skills</div>
   976→                </div>
   977→              )}
   978→            </div>
   979→            <div className="px-4 py-2 border-t border-[var(--j-border)] bg-[var(--j-panel-soft)]/40 flex items-center justify-between jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">
   980→              <span>{results.length} results</span>
   981→              <span className="flex items-center gap-2">
   982→                <span>↑↓ navigate</span>
   983→                <span>⏎ open</span>
   984→              </span>
   985→            </div>
   986→          </motion.div>
   987→        </motion.div>
   988→      )}
   989→    </AnimatePresence>
   990→  );
   991→}
