   189→

   193→    config: {
   194→      appName: string;
   195→      version: string;

   217→      setTheme(saved);
   218→      document.documentElement.classList.toggle('light', saved === 'light');

   221→  }, []);

   223→  // Tab prefs: load persisted personalization on mount.
   224→  useEffect(() => {
   225→    try {
   226→      const raw = localStorage.getItem('jarvis-tab-prefs');
   227→      if (raw) {
   228→        const p = JSON.parse(raw);
   229→        setTabPrefs({ hidden: p.hidden ?? [], pinned: p.pinned ?? [], order: p.order ?? [] });
   230→      }
   231→    } catch { /* ignore */ }
   232→  }, []);

   235→    setTabPrefs((prev) => {
   236→      const next = updater(prev);
   237→      try { localStorage.setItem('jarvis-tab-prefs', JSON.stringify(next)); } catch { /* ignore */ }
   238→      return next;
   239→    });
   240→  }, []);

   243→    setTheme((t) => {
   244→      const next = t === 'dark' ? 'light' : 'dark';
   245→      document.documentElement.classList.toggle('light', next === 'light');
   246→      document.documentElement.classList.toggle('dark', next === 'dark');
   247→      try { localStorage.setItem('jarvis-theme', next); } catch { /* ignore */ }
   248→      return next;
   249→    });
   250→  }, []);
   251→
   252→  // Live clock — only after mount to avoid hydration mismatch.
   253→  useEffect(() => {

   261→    setPaletteOpen(true);

   268→        e.preventDefault();
   269→        setPaletteOpen((o) => {
   270→          if (!o) setPaletteKey((k) => k + 1);
   271→          return !o;
   272→        });
   273→      }
   274→      if ((e.metaKey || e.ctrlKey) && e.shiftKey && e.key.toLowerCase() === 'f') {
   275→        e.preventDefault();
   276→        setGlobalSearchOpen((o) => !o);
   277→      }
   278→      if (e.key === 'Escape') {
   279→        setPaletteOpen(false);
   280→        setGlobalSearchOpen(false);
   281→        setNotifOpen(false);
   282→      }

   303→        e.preventDefault();
   304→        setOrionMode((o) => !o);
   305→      }

   329→  const cur = TABS.find((t) => t.key === tab)!;

   676→/* ---------- Command palette ---------- */
   677→function CommandPalette({ open, onClose, onNavigate }: { open: boolean; onClose: () => void; onNavigate: (t: TabKey) => void }) {
   678→  // Fresh mount (via key in parent) ensures q/sel start empty each open.
   679→  const [q, setQ] = useState('');
   680→  const [sel, setSel] = useState(0);
   681→
   682→  const results = useMemo(() => {
   683→    const ql = q.toLowerCase();
   684→    return TABS.filter((t) => !ql || t.label.toLowerCase().includes(ql) || t.group.toLowerCase().includes(ql));
   685→  }, [q]);
   686→
   687→  useEffect(() => {
   688→    if (!open) return;
   689→    const onKey = (e: KeyboardEvent) => {
   690→      if (e.key === 'ArrowDown') { e.preventDefault(); setSel((s) => Math.min(s + 1, results.length - 1)); }
   691→      if (e.key === 'ArrowUp') { e.preventDefault(); setSel((s) => Math.max(s - 1, 0)); }
   692→      if (e.key === 'Enter') { e.preventDefault(); const r = results[sel]; if (r) onNavigate(r.key); }
   693→    };
   694→    window.addEventListener('keydown', onKey);
   695→    return () => window.removeEventListener('keydown', onKey);
   696→  }, [open, results, sel, onNavigate]);
   697→
   698→  return (
   699→    <AnimatePresence>
   700→      {open && (
   701→        <motion.div
   702→          className="fixed inset-0 z-50 flex items-start justify-center pt-[12vh] px-4"
   703→          initial={{ opacity: 0 }}
   704→          animate={{ opacity: 1 }}
   705→          exit={{ opacity: 0 }}
   706→        >
   707→          <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   708→          <motion.div
   709→            initial={{ opacity: 0, scale: 0.96, y: -8 }}
   710→            animate={{ opacity: 1, scale: 1, y: 0 }}
   711→            exit={{ opacity: 0, scale: 0.96, y: -8 }}
   712→            transition={{ duration: 0.18 }}
   713→            className="relative w-full max-w-xl jarvis-panel p-0 overflow-hidden"
   714→          >
   715→            <div className="flex items-center gap-2.5 px-4 h-12 border-b border-[var(--j-border)]">
   716→              <Search className="h-4 w-4 text-[var(--j-text-mute)]" />
   717→              <input
   718→                autoFocus
   719→                value={q}
   720→                onChange={(e) => { setQ(e.target.value); setSel(0); }}
   721→                placeholder="Search tabs and actions…"
   722→                className="flex-1 bg-transparent outline-none text-sm text-[var(--j-text)] placeholder:text-[var(--j-text-mute)]"
   723→              />
   724→              <kbd className="jarvis-mono text-[9px] px-1.5 py-0.5 rounded border border-[var(--j-border)] text-[var(--j-text-mute)]">ESC</kbd>
   725→            </div>
   726→            <div className="max-h-72 overflow-y-auto jarvis-scroll p-2">
   727→              {results.length ? results.map((t, i) => {
   728→                const Icon = t.icon;
   729→                const active = i === sel;
   730→                return (
   731→                  <button
   732→                    key={t.key}
   733→                    onMouseEnter={() => setSel(i)}
   734→                    onClick={() => onNavigate(t.key)}
   735→                    className={cn('w-full flex items-center gap-3 px-3 py-2 rounded-md text-sm transition-colors', active ? 'bg-[var(--j-panel-soft)] text-[var(--j-text)]' : 'text-[var(--j-text-dim)]')}
   736→                  >
   737→                    <Icon className="h-4 w-4" style={{ color: active ? t.accent : undefined }} />
   738→                    <span className="flex-1 text-left">{t.label}</span>
   739→                    <span className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">{t.group}</span>
   740→                  </button>
   741→                );
   742→              }) : (
   743→                <div className="px-3 py-8 text-center text-xs text-[var(--j-text-mute)]">No results</div>
   744→              )}
   745→            </div>
   746→          </motion.div>
   747→        </motion.div>
   748→      )}
   749→    </AnimatePresence>
   750→  );
   751→}

   753→/* ---------- Manage Tabs modal (personalization) ---------- */
   754→function ManageTabsModal({
   755→  onClose,
   756→  tabPrefs,
   757→  updateTabPrefs,
   758→  onNavigate,
   759→}: {
   760→  onClose: () => void;
   761→  tabPrefs: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] };
   762→  updateTabPrefs: (updater: (prev: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => void;
   763→  onNavigate: (t: TabKey) => void;
   764→}) {
   765→  const toggleHidden = (key: TabKey) => {
   766→    updateTabPrefs((prev) => ({
   767→      ...prev,
   768→      hidden: prev.hidden.includes(key) ? prev.hidden.filter((k) => k !== key) : [...prev.hidden, key],
   769→    }));
   770→  };
   771→  const togglePinned = (key: TabKey) => {
   772→    updateTabPrefs((prev) => ({
   773→      ...prev,
   774→      pinned: prev.pinned.includes(key) ? prev.pinned.filter((k) => k !== key) : [...prev.pinned, key],
   775→    }));
   776→  };
   777→  const resetAll = () => {
   778→    updateTabPrefs(() => ({ hidden: [], pinned: [], order: [] }));
   779→  };
   780→  const visibleCount = TABS.length - tabPrefs.hidden.length;
   781→
   782→  return (
   783→    <motion.div className="fixed inset-0 z-50 flex items-center justify-center p-4" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }}>
   784→      <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   785→      <motion.div initial={{ scale: 0.96, opacity: 0 }} animate={{ scale: 1, opacity: 1 }} exit={{ scale: 0.96, opacity: 0 }} className="relative w-full max-w-2xl jarvis-panel p-0 overflow-hidden max-h-[85vh] flex flex-col">
   786→        <div className="flex items-center justify-between px-5 py-3 border-b border-[var(--j-border)]">
   787→          <div className="flex items-center gap-2">
   788→            <Sliders className="h-4 w-4 text-[var(--j-cyan)]" />
   789→            <h3 className="jarvis-mono text-sm uppercase text-[var(--j-cyan)]">Manage Tabs</h3>
   790→            <span className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">{visibleCount} visible · {tabPrefs.pinned.length} pinned · {tabPrefs.hidden.length} hidden</span>
   791→          </div>
   792→          <div className="flex items-center gap-2">
   793→            <button onClick={resetAll} className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)] hover:text-[var(--j-cyan)]">reset</button>
   794→            <button onClick={onClose} className="text-[var(--j-text-mute)] hover:text-[var(--j-text)]"><X className="h-4 w-4" /></button>
   795→          </div>
   796→        </div>
   797→        <div className="overflow-y-auto jarvis-scroll p-3">
   798→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-2">
   799→            {TABS.map((t) => {
   800→              const isHidden = tabPrefs.hidden.includes(t.key);
   801→              const isPinned = tabPrefs.pinned.includes(t.key);
   802→              const Icon = t.icon;
   803→              return (
   804→                <div
   805→                  key={t.key}
   806→                  className={`flex items-center gap-2.5 p-2.5 rounded-lg border transition-all ${isHidden ? 'border-[var(--j-border-soft)] opacity-50' : 'border-[var(--j-border)] bg-[var(--j-panel-soft)]/40'}`}
   807→                >
   808→                  <button
   809→                    onClick={() => togglePinned(t.key)}
   810→                    className={`flex h-7 w-7 items-center justify-center rounded-md shrink-0 transition-colors ${isPinned ? 'text-[var(--j-violet)]' : 'text-[var(--j-text-mute)] hover:text-[var(--j-violet)]'}`}
   811→                    title={isPinned ? 'Unpin' : 'Pin to top'}
   812→                  >
   813→                    <Pin className={`h-3.5 w-3.5 ${isPinned ? 'fill-[var(--j-violet)]' : ''}`} />
   814→                  </button>
   815→                  <div className="flex h-7 w-7 items-center justify-center rounded-md shrink-0" style={{ background: `${t.accent}1a`, border: `1px solid ${t.accent}33`, color: t.accent }}>
   816→                    <Icon className="h-3.5 w-3.5" />
   817→                  </div>
   818→                  <button onClick={() => !isHidden && onNavigate(t.key)} className="flex-1 text-left min-w-0">
   819→                    <div className="text-xs text-[var(--j-text)] truncate">{t.label}</div>
   820→                    <div className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">{isPinned ? 'pinned · ' : ''}{t.group}</div>
   821→                  </button>
   822→                  <button
   823→                    onClick={() => toggleHidden(t.key)}
   824→                    className={`flex h-7 w-7 items-center justify-center rounded-md shrink-0 transition-colors ${isHidden ? 'text-[var(--j-text-mute)] hover:text-[var(--j-green)]' : 'text-[var(--j-green)] hover:text-[var(--j-red)]'}`}
   825→                    title={isHidden ? 'Show' : 'Hide'}
   826→                  >
   827→                    {isHidden ? <EyeOff className="h-3.5 w-3.5" /> : <Eye className="h-3.5 w-3.5" />}
   828→                  </button>
   829→                </div>
   830→              );
   831→            })}
   832→          </div>
   833→        </div>
   834→        <div className="px-5 py-2.5 border-t border-[var(--j-border)] bg-[var(--j-panel-soft)]/40 flex items-center justify-between">
   835→          <span className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">preferences saved to this browser</span>
   836→          <button onClick={onClose} className="jarvis-mono text-[10px] uppercase px-3 py-1 rounded jarvis-btn-accent border-0">Done</button>
   837→        </div>
   838→      </motion.div>
   839→    </motion.div>

   843→/* ---------- Global search overlay (Cmd+Shift+F) ---------- */
   844→interface SearchResult {
   845→  id: string;
   846→  type: 'agent' | 'task' | 'memory' | 'comms' | 'skill';
   847→  title: string;
   848→  subtitle: string;
   849→  meta: string;
   850→  color: string;
   851→  href: TabKey;
   852→  score: number;
   853→}
   854→
   855→function GlobalSearch({ open, onClose, onNavigate }: { open: boolean; onClose: () => void; onNavigate: (t: TabKey) => void }) {
   856→  const [q, setQ] = useState('');
   857→  const [results, setResults] = useState<SearchResult[]>([]);
   858→  const [loading, setLoading] = useState(false);
   859→  const [sel, setSel] = useState(0);
   860→  const [savedSearches, setSavedSearches] = useState<string[]>([]);
   861→
   862→  // Load saved searches from localStorage on mount.
   863→  useEffect(() => {
   864→    try {
   865→      const raw = localStorage.getItem('jarvis-saved-searches');
   866→      if (raw) setSavedSearches(JSON.parse(raw));
   867→    } catch { /* ignore */ }
   868→  }, []);
   869→
   870→  const saveSearch = (term: string) => {
   871→    const t = term.trim();
   872→    if (!t) return;
   873→    setSavedSearches((prev) => {
   874→      const next = prev.includes(t) ? prev.filter((s) => s !== t) : [...prev, t].slice(-8);
   875→      try { localStorage.setItem('jarvis-saved-searches', JSON.stringify(next)); } catch { /* ignore */ }
   876→      return next;
   877→    });
   878→  };
   879→
   880→  // Debounced search.
   881→  useEffect(() => {
   882→    if (!open) { setQ(''); setResults([]); setSel(0); return; }
   883→    if (!q.trim()) { setResults([]); return; }
   884→    setLoading(true);
   885→    const id = setTimeout(async () => {
   886→      try {
   887→        const res = await fetch(`/api/search?q=${encodeURIComponent(q)}`, { cache: 'no-store' });
   888→        const json = await res.json();
   889→        setResults(json.results ?? []);
   890→        setSel(0);
   891→      } catch { /* ignore */ } finally {
   892→        setLoading(false);
   893→      }
   894→    }, 200);
   895→    return () => clearTimeout(id);
   896→  }, [q, open]);
   897→
   898→  useEffect(() => {
   899→    if (!open) return;
   900→    const onKey = (e: KeyboardEvent) => {
   901→      if (e.key === 'ArrowDown') { e.preventDefault(); setSel((s) => Math.min(s + 1, results.length - 1)); }
   902→      if (e.key === 'ArrowUp') { e.preventDefault(); setSel((s) => Math.max(s - 1, 0)); }
   903→      if (e.key === 'Enter') { e.preventDefault(); const r = results[sel]; if (r) { onNavigate(r.href); onClose(); } }
   904→    };
   905→    window.addEventListener('keydown', onKey);
   906→    return () => window.removeEventListener('keydown', onKey);
   907→  }, [open, results, sel, onNavigate, onClose]);
   908→
   909→  const typeIcon: Record<string, typeof Bot> = { agent: Bot, task: ListTodo, memory: Database, comms: MessagesSquare, skill: Sparkles };
   910→
   911→  return (
   912→    <AnimatePresence>
   913→      {open && (
   914→        <motion.div className="fixed inset-0 z-50 flex items-start justify-center pt-[10vh] px-4" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }}>
   915→          <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   916→          <motion.div initial={{ scale: 0.96, opacity: 0, y: -8 }} animate={{ scale: 1, opacity: 1, y: 0 }} exit={{ scale: 0.96, opacity: 0, y: -8 }} className="relative w-full max-w-2xl jarvis-panel p-0 overflow-hidden">
   917→            <div className="flex items-center gap-2.5 px-4 h-12 border-b border-[var(--j-border)]">
   918→              <Search className="h-4 w-4 text-[var(--j-cyan)]" />
   919→              <input
   920→                autoFocus
   921→                value={q}
   922→                onChange={(e) => setQ(e.target.value)}
   923→                placeholder="Search agents, tasks, memory, comms, skills…"
   924→                className="flex-1 bg-transparent outline-none text-sm text-[var(--j-text)] placeholder:text-[var(--j-text-mute)]"
   925→              />
   926→              {loading && <Loader2 className="h-3.5 w-3.5 animate-spin text-[var(--j-cyan)]" />}
   927→              {q.trim() && (
   928→                <button onClick={() => saveSearch(q)} className="flex items-center gap-1 jarvis-mono text-[9px] uppercase text-[var(--j-cyan)] hover:underline shrink-0">
   929→                  <Star className="h-3 w-3" /> {savedSearches.includes(q.trim()) ? 'saved' : 'save'}
   930→                </button>
   931→              )}
   932→              <kbd className="jarvis-mono text-[9px] px-1.5 py-0.5 rounded border border-[var(--j-border)] text-[var(--j-text-mute)]">ESC</kbd>
   933→            </div>
   934→            <div className="max-h-96 overflow-y-auto jarvis-scroll p-2">
   935→              {results.length > 0 ? (
   936→                results.map((r, i) => {
   937→                  const Icon = typeIcon[r.type] ?? Search;
   938→                  const active = i === sel;
   939→                  return (
   940→                    <button
   941→                      key={r.id}
   942→                      onMouseEnter={() => setSel(i)}
   943→                      onClick={() => { onNavigate(r.href); onClose(); }}
   944→                      className={cn('w-full flex items-center gap-3 px-3 py-2.5 rounded-md text-left transition-colors', active ? 'bg-[var(--j-panel-soft)]' : 'hover:bg-[var(--j-panel-soft)]/60')}
   945→                    >
   946→                      <div className="flex h-8 w-8 items-center justify-center rounded-md shrink-0" style={{ background: `${r.color}1a`, border: `1px solid ${r.color}33`, color: r.color }}>
   947→                        <Icon className="h-4 w-4" />
   948→                      </div>
   949→                      <div className="min-w-0 flex-1">
   950→                        <div className="text-sm text-[var(--j-text)] truncate">{r.title}</div>
   951→                        <div className="text-[11px] text-[var(--j-text-dim)] truncate">{r.subtitle}</div>
   952→                      </div>
   953→                      <div className="text-right shrink-0">
   954→                        <div className="jarvis-mono text-[9px] uppercase px-1.5 py-0.5 rounded" style={{ color: r.color, background: `${r.color}1a` }}>{r.type}</div>
   955→                        <div className="jarvis-mono text-[9px] text-[var(--j-text-mute)] mt-0.5">{r.meta}</div>
   956→                      </div>
   957→                    </button>
   958→                  );
   959→                })
   960→              ) : q.trim() ? (
   961→                <div className="px-3 py-10 text-center">
   962→                  <Search className="h-8 w-8 mx-auto mb-2 text-[var(--j-text-mute)] opacity-40" />
   963→                  <div className="text-sm text-[var(--j-text-mute)]">No results for "{q}"</div>
   964→                </div>
   965→              ) : savedSearches.length > 0 ? (
   966→                <div className="px-3 py-4">
   967→                  <div className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)] mb-2 flex items-center gap-1.5"><Star className="h-3 w-3" /> saved searches</div>
   968→                  <div className="flex flex-wrap gap-2">
   969→                    {savedSearches.map((s) => (
   970→                      <button key={s} onClick={() => setQ(s)} className="group flex items-center gap-1.5 text-xs px-2.5 py-1.5 rounded-md border border-[var(--j-border)] bg-[var(--j-panel-soft)] text-[var(--j-text-dim)] hover:border-[var(--j-cyan)] hover:text-[var(--j-cyan)] transition-colors">
   971→                        <Search className="h-3 w-3" /> {s}
   972→                        <span onClick={(e) => { e.stopPropagation(); saveSearch(s); }} className="text-[var(--j-text-mute)] hover:text-[var(--j-red)] ml-0.5"><X className="h-2.5 w-2.5" /></span>
   973→                      </button>
   974→                    ))}
   975→                  </div>
   976→                </div>
   977→              ) : (
   978→                <div className="px-3 py-10 text-center">
   979→                  <div className="text-sm text-[var(--j-text-mute)]">Type to search across the fleet</div>
   980→                  <div className="jarvis-mono text-[10px] text-[var(--j-text-mute)] mt-1">agents · tasks · memory · comms · skills</div>
   981→                </div>
   982→              )}
   983→            </div>
   984→            <div className="px-4 py-2 border-t border-[var(--j-border)] bg-[var(--j-panel-soft)]/40 flex items-center justify-between jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">
   985→              <span>{results.length} results</span>
   986→              <span className="flex items-center gap-2">
   987→                <span>↑↓ navigate</span>
   988→                <span>⏎ open</span>
   989→              </span>
   990→            </div>
   991→          </motion.div>
   992→        </motion.div>
   993→      )}
   994→    </AnimatePresence>
   995→  );
   996→}
