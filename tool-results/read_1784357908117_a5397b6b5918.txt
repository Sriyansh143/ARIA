   173→export default function MissionControlDashboard() {
   174→  // Single source of truth for the active tab is the global nav store,
   175→  // so any component (StatCard, Orion voice command, agent alerts, …) can
   176→  // navigate without prop-drilling. The shell subscribes here.
   177→  const tab = useNavStore((s) => s.tab as TabKey);
   178→  const navigateStore = useNavStore((s) => s.navigate);
   179→  const [sidebarOpen, setSidebarOpen] = useState(false);
   180→  const [paletteOpen, setPaletteOpen] = useState(false);
   181→  const [paletteKey, setPaletteKey] = useState(0);
   182→  const [globalSearchOpen, setGlobalSearchOpen] = useState(false);
   183→  const [notifOpen, setNotifOpen] = useState(false);
   184→  const [now, setNow] = useState<Date | null>(null);
   185→  const [theme, setTheme] = useState<'dark' | 'light'>('dark');
   186→  const [manageOpen, setManageOpen] = useState(false);
   187→  // Tab personalization: hidden tabs + pinned tabs + custom order.
   188→  const [tabPrefs, setTabPrefs] = useState<{ hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }>({ hidden: [], pinned: [], order: [] });
   189→  // Orion voice-shell overlay mode (persisted). When true, a full-screen
   190→  // voice-first interface replaces the dashboard.
   191→  const [orionMode, setOrionMode] = useState(false);
   192→
   193→  const { data: metrics } = useApi<MetricsData>('/api/metrics', 8000);
   194→  const { data: commsData } = useApi<{ unread: number }>('/api/comms', 15000);
   195→  const commsUnread = commsData?.unread ?? 0;
   196→
   197→  // Branding — fetch once on mount (-1 = no polling). Drives the header
   198→  // app-name, version, footer powered-by, and chat-tab label.
   199→  const { data: brandingData } = useApi<{
   200→    config: {
   201→      appName: string;
   202→      version: string;
   203→      poweredBy?: string;
   204→      chatTabLabel?: string;
   205→      footerNote?: string;
   206→    };
   207→    defaults: { appName: string; version: string };
   208→  }>('/api/branding', -1);
   209→  const branding = brandingData?.config;
   210→  const appName = branding?.appName || 'JARVIS';
   211→  const appVersion = branding?.version || brandingData?.defaults.version || JARVIS.version;
   212→  const appPoweredBy = branding?.poweredBy || '';
   213→  const appCompany = branding?.company || 'Liafon Software Private Limited';
   214→  const chatTabLabel = branding?.chatTabLabel || 'JARVIS Chat';
   215→  const footerNote = branding?.footerNote || '';
   216→
   217→  // Resolve a tab's display label — chat label is driven by branding config.
   218→  const tabLabelOf = useCallback((t: TabDef) => (t.key === 'chat' ? chatTabLabel : t.label), [chatTabLabel]);
   219→
   220→  // Theme: load persisted preference on mount, default dark.
   221→  useEffect(() => {
   222→    const saved = (typeof window !== 'undefined' && localStorage.getItem('jarvis-theme')) as 'dark' | 'light' | null;
   223→    if (saved === 'light' || saved === 'dark') {
   224→      setTheme(saved);
   225→      document.documentElement.classList.toggle('light', saved === 'light');
   226→      document.documentElement.classList.toggle('dark', saved === 'dark');
   227→    }
   228→  }, []);
   229→
   230→  // Tab prefs: load persisted personalization on mount.
   231→  useEffect(() => {
   232→    try {
   233→      const raw = localStorage.getItem('jarvis-tab-prefs');
   234→      if (raw) {
   235→        const p = JSON.parse(raw);
   236→        setTabPrefs({ hidden: p.hidden ?? [], pinned: p.pinned ?? [], order: p.order ?? [] });
   237→      }
   238→    } catch { /* ignore */ }
   239→  }, []);
   240→
   241→  const updateTabPrefs = useCallback((updater: (prev: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => {
   242→    setTabPrefs((prev) => {
   243→      const next = updater(prev);
   244→      try { localStorage.setItem('jarvis-tab-prefs', JSON.stringify(next)); } catch { /* ignore */ }
   245→      return next;
   246→    });
   247→  }, []);
   248→
   249→  const toggleTheme = useCallback(() => {
   250→    setTheme((t) => {
   251→      const next = t === 'dark' ? 'light' : 'dark';
   252→      document.documentElement.classList.toggle('light', next === 'light');
   253→      document.documentElement.classList.toggle('dark', next === 'dark');
   254→      try { localStorage.setItem('jarvis-theme', next); } catch { /* ignore */ }
   255→      return next;
   256→    });
   257→  }, []);
   258→
   259→  // Live clock — only after mount to avoid hydration mismatch.
   260→  useEffect(() => {
   261→    setNow(new Date());
   262→    const id = setInterval(() => setNow(new Date()), 1000);
   263→    return () => clearInterval(id);
   264→  }, []);
   265→
   266→  const openPalette = useCallback(() => {
   267→    setPaletteKey((k) => k + 1);
   268→    setPaletteOpen(true);
   269→  }, []);
   270→
   271→  // Command palette shortcut (Cmd/Ctrl+K).
   272→  useEffect(() => {
   273→    const onKey = (e: KeyboardEvent) => {
   274→      if ((e.metaKey || e.ctrlKey) && e.key.toLowerCase() === 'k') {
   275→        e.preventDefault();
   276→        setPaletteOpen((o) => {
   277→          if (!o) setPaletteKey((k) => k + 1);
   278→          return !o;
   279→        });
   280→      }
   281→      if ((e.metaKey || e.ctrlKey) && e.shiftKey && e.key.toLowerCase() === 'f') {
   282→        e.preventDefault();
   283→        setGlobalSearchOpen((o) => !o);
   284→      }
   285→      if (e.key === 'Escape') {
   286→        setPaletteOpen(false);
   287→        setGlobalSearchOpen(false);
   288→        setNotifOpen(false);
   289→      }
   290→    };
   291→    window.addEventListener('keydown', onKey);
   292→    return () => window.removeEventListener('keydown', onKey);
   293→  }, []);
   294→
   295→  // Orion mode: load persisted preference on mount, then persist on change.
   296→  useEffect(() => {
   297→    try {
   298→      const raw = localStorage.getItem('jarvis-orion-mode');
   299→      if (raw === '1') setOrionMode(true);
   300→    } catch { /* ignore */ }
   301→  }, []);
   302→  useEffect(() => {
   303→    try { localStorage.setItem('jarvis-orion-mode', orionMode ? '1' : '0'); } catch { /* ignore */ }
   304→  }, [orionMode]);
   305→
   306→  // Orion mode shortcut (Cmd/Ctrl+Shift+O).
   307→  useEffect(() => {
   308→    const onKey = (e: KeyboardEvent) => {
   309→      if ((e.metaKey || e.ctrlKey) && e.shiftKey && e.key.toLowerCase() === 'o') {
   310→        e.preventDefault();
   311→        setOrionMode((o) => !o);
   312→      }
   313→    };
   314→    window.addEventListener('keydown', onKey);
   315→    return () => window.removeEventListener('keydown', onKey);
   316→  }, []);
   317→
   318→  const navigate = useCallback((t: TabKey) => {
   319→    navigateStore(t);
   320→    setSidebarOpen(false);
   321→    setPaletteOpen(false);
   322→  }, [navigateStore]);
   323→
   324→  const ActiveTab = TAB_MAP[tab];
   325→
   326→  const grouped = useMemo(() => {
   327→    const visibleTabs = TABS.filter((t) => !tabPrefs.hidden.includes(t.key));
   328→    const pinnedTabs = visibleTabs.filter((t) => tabPrefs.pinned.includes(t.key));
   329→    const unpinnedTabs = visibleTabs.filter((t) => !tabPrefs.pinned.includes(t.key));
   330→    const g: Record<string, TabDef[]> = {};
   331→    if (pinnedTabs.length) g['Pinned'] = pinnedTabs;
   332→    for (const t of unpinnedTabs) (g[t.group] ??= []).push(t);
   333→    return g;
   334→  }, [tabPrefs.hidden, tabPrefs.pinned]);
   335→
   336→  const cur = TABS.find((t) => t.key === tab)!;
   337→
   338→  return (
   339→    <div className="min-h-screen flex flex-col relative">
   340→      <div className="jarvis-bg" />
   341→
   342→      {/* Header */}
   343→      <header className="sticky top-0 z-40 border-b border-[var(--j-border)] bg-[rgba(7,8,10,0.82)] backdrop-blur-xl">
   344→        <div className="flex items-center gap-3 px-4 h-14">
   345→          <button
   346→            className="lg:hidden flex h-8 w-8 items-center justify-center rounded-md border border-[var(--j-border)] text-[var(--j-text-dim)] hover:text-[var(--j-cyan)]"
   347→            onClick={() => setSidebarOpen((o) => !o)}
   348→            aria-label="Toggle sidebar"
   349→          >
   350→            {sidebarOpen ? <X className="h-4 w-4" /> : <Menu className="h-4 w-4" />}
   351→          </button>
   352→
   353→          <div className="flex items-center gap-2.5">
   354→            <div className="relative">
   355→              <div className="flex h-8 w-8 items-center justify-center rounded-md jarvis-btn-accent">
   356→                <ShieldCheck className="h-4 w-4" />
   357→              </div>
   358→              <CircleDot className="absolute -top-1 -right-1 h-2.5 w-2.5 text-[var(--j-green)] jarvis-blink" />
   359→            </div>
   360→            <div className="leading-none">
   361→              <div className="text-sm font-bold tracking-tight jarvis-text-gradient">{appName}</div>
   362→              <div className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">Mission Control v{appVersion}</div>
   363→            </div>
   364→          </div>
   365→
   366→          {/* Mini metrics */}
   367→          <div className="hidden md:flex items-center gap-3 ml-4">
   368→            <MiniMetric icon={Cpu} label="CPU" value={metrics ? `${Math.round(metrics.current.cpu)}%` : '—'} color={JARVIS.colors.cyan} />
   369→            <MiniMetric icon={MemoryStick} label="MEM" value={metrics ? `${Math.round(metrics.current.mem)}%` : '—'} color={JARVIS.colors.violet} />
   370→            <MiniMetric icon={Zap} label="LAT" value={metrics ? `${metrics.current.latency}ms` : '—'} color={JARVIS.colors.amber} />

   372→          </div>
   373→
   374→          <div className="flex-1" />
   375→
   376→          {/* Live clock */}
   377→          <div className="hidden sm:flex flex-col items-end leading-none mr-1">
   378→            <div className="jarvis-mono text-sm text-[var(--j-cyan)] tabular-nums">{now ? fmtTime(now) : '--:--:--'}</div>

   382→          <button

   385→          >

   390→
   391→          <button
   392→            onClick={toggleTheme}
   393→            className="flex h-8 w-8 items-center justify-center rounded-md border border-[var(--j-border)] bg-[var(--j-panel-soft)] text-[var(--j-text-dim)] hover:border-[var(--j-cyan)] hover:text-[var(--j-cyan)] transition-colors"
   394→            aria-label="Toggle theme"
   395→            title={theme === 'dark' ? 'Switch to light mode' : 'Switch to dark mode'}

   397→            {theme === 'dark' ? <Sun className="h-3.5 w-3.5" /> : <Moon className="h-3.5 w-3.5" />}

   399→
   400→          <NotificationsBell open={notifOpen} setOpen={setNotifOpen} />
   401→        </div>
   402→        <div className="h-px jarvis-hr" />

   404→
   405→      <div className="flex flex-1 relative z-10">
   406→        {/* Sidebar */}
   407→        <AnimatePresence>
   408→          {sidebarOpen && (
   409→            <motion.div

   683→/* ---------- Command palette ---------- */
   684→function CommandPalette({ open, onClose, onNavigate }: { open: boolean; onClose: () => void; onNavigate: (t: TabKey) => void }) {
   685→  // Fresh mount (via key in parent) ensures q/sel start empty each open.
   686→  const [q, setQ] = useState('');
   687→  const [sel, setSel] = useState(0);
   688→
   689→  const results = useMemo(() => {
   690→    const ql = q.toLowerCase();
   691→    return TABS.filter((t) => !ql || t.label.toLowerCase().includes(ql) || t.group.toLowerCase().includes(ql));
   692→  }, [q]);
   693→
   694→  useEffect(() => {
   695→    if (!open) return;
   696→    const onKey = (e: KeyboardEvent) => {
   697→      if (e.key === 'ArrowDown') { e.preventDefault(); setSel((s) => Math.min(s + 1, results.length - 1)); }
   698→      if (e.key === 'ArrowUp') { e.preventDefault(); setSel((s) => Math.max(s - 1, 0)); }
   699→      if (e.key === 'Enter') { e.preventDefault(); const r = results[sel]; if (r) onNavigate(r.key); }
   700→    };
   701→    window.addEventListener('keydown', onKey);
   702→    return () => window.removeEventListener('keydown', onKey);
   703→  }, [open, results, sel, onNavigate]);
   704→
   705→  return (
   706→    <AnimatePresence>
   707→      {open && (
   708→        <motion.div
   709→          className="fixed inset-0 z-50 flex items-start justify-center pt-[12vh] px-4"
   710→          initial={{ opacity: 0 }}
   711→          animate={{ opacity: 1 }}
   712→          exit={{ opacity: 0 }}
   713→        >
   714→          <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   715→          <motion.div
   716→            initial={{ opacity: 0, scale: 0.96, y: -8 }}
   717→            animate={{ opacity: 1, scale: 1, y: 0 }}
   718→            exit={{ opacity: 0, scale: 0.96, y: -8 }}
   719→            transition={{ duration: 0.18 }}
   720→            className="relative w-full max-w-xl jarvis-panel p-0 overflow-hidden"
   721→          >
   722→            <div className="flex items-center gap-2.5 px-4 h-12 border-b border-[var(--j-border)]">
   723→              <Search className="h-4 w-4 text-[var(--j-text-mute)]" />
   724→              <input
   725→                autoFocus
   726→                value={q}
   727→                onChange={(e) => { setQ(e.target.value); setSel(0); }}
   728→                placeholder="Search tabs and actions…"
   729→                className="flex-1 bg-transparent outline-none text-sm text-[var(--j-text)] placeholder:text-[var(--j-text-mute)]"
   730→              />
   731→              <kbd className="jarvis-mono text-[9px] px-1.5 py-0.5 rounded border border-[var(--j-border)] text-[var(--j-text-mute)]">ESC</kbd>
   732→            </div>
   733→            <div className="max-h-72 overflow-y-auto jarvis-scroll p-2">
   734→              {results.length ? results.map((t, i) => {
   735→                const Icon = t.icon;
   736→                const active = i === sel;
   737→                return (
   738→                  <button
   739→                    key={t.key}
   740→                    onMouseEnter={() => setSel(i)}
   741→                    onClick={() => onNavigate(t.key)}
   742→                    className={cn('w-full flex items-center gap-3 px-3 py-2 rounded-md text-sm transition-colors', active ? 'bg-[var(--j-panel-soft)] text-[var(--j-text)]' : 'text-[var(--j-text-dim)]')}
   743→                  >
   744→                    <Icon className="h-4 w-4" style={{ color: active ? t.accent : undefined }} />
   745→                    <span className="flex-1 text-left">{t.label}</span>
   746→                    <span className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">{t.group}</span>
   747→                  </button>
   748→                );
   749→              }) : (
   750→                <div className="px-3 py-8 text-center text-xs text-[var(--j-text-mute)]">No results</div>
   751→              )}
   752→            </div>
   753→          </motion.div>
   754→        </motion.div>
   755→      )}
   756→    </AnimatePresence>
   757→  );
   758→}

   760→/* ---------- Manage Tabs modal (personalization) ---------- */
   761→function ManageTabsModal({
   762→  onClose,
   763→  tabPrefs,
   764→  updateTabPrefs,
   765→  onNavigate,
   766→}: {
   767→  onClose: () => void;
   768→  tabPrefs: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] };
   769→  updateTabPrefs: (updater: (prev: { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => { hidden: TabKey[]; pinned: TabKey[]; order: TabKey[] }) => void;
   770→  onNavigate: (t: TabKey) => void;
   771→}) {
   772→  const toggleHidden = (key: TabKey) => {
   773→    updateTabPrefs((prev) => ({
   774→      ...prev,
   775→      hidden: prev.hidden.includes(key) ? prev.hidden.filter((k) => k !== key) : [...prev.hidden, key],
   776→    }));
   777→  };
   778→  const togglePinned = (key: TabKey) => {
   779→    updateTabPrefs((prev) => ({
   780→      ...prev,
   781→      pinned: prev.pinned.includes(key) ? prev.pinned.filter((k) => k !== key) : [...prev.pinned, key],
   782→    }));
   783→  };
   784→  const resetAll = () => {
   785→    updateTabPrefs(() => ({ hidden: [], pinned: [], order: [] }));
   786→  };
   787→  const visibleCount = TABS.length - tabPrefs.hidden.length;
   788→
   789→  return (
   790→    <motion.div className="fixed inset-0 z-50 flex items-center justify-center p-4" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }}>
   791→      <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   792→      <motion.div initial={{ scale: 0.96, opacity: 0 }} animate={{ scale: 1, opacity: 1 }} exit={{ scale: 0.96, opacity: 0 }} className="relative w-full max-w-2xl jarvis-panel p-0 overflow-hidden max-h-[85vh] flex flex-col">
   793→        <div className="flex items-center justify-between px-5 py-3 border-b border-[var(--j-border)]">
   794→          <div className="flex items-center gap-2">
   795→            <Sliders className="h-4 w-4 text-[var(--j-cyan)]" />
   796→            <h3 className="jarvis-mono text-sm uppercase text-[var(--j-cyan)]">Manage Tabs</h3>
   797→            <span className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">{visibleCount} visible · {tabPrefs.pinned.length} pinned · {tabPrefs.hidden.length} hidden</span>
   798→          </div>
   799→          <div className="flex items-center gap-2">
   800→            <button onClick={resetAll} className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)] hover:text-[var(--j-cyan)]">reset</button>
   801→            <button onClick={onClose} className="text-[var(--j-text-mute)] hover:text-[var(--j-text)]"><X className="h-4 w-4" /></button>
   802→          </div>
   803→        </div>
   804→        <div className="overflow-y-auto jarvis-scroll p-3">
   805→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-2">
   806→            {TABS.map((t) => {
   807→              const isHidden = tabPrefs.hidden.includes(t.key);
   808→              const isPinned = tabPrefs.pinned.includes(t.key);
   809→              const Icon = t.icon;
   810→              return (
   811→                <div
   812→                  key={t.key}
   813→                  className={`flex items-center gap-2.5 p-2.5 rounded-lg border transition-all ${isHidden ? 'border-[var(--j-border-soft)] opacity-50' : 'border-[var(--j-border)] bg-[var(--j-panel-soft)]/40'}`}
   814→                >
   815→                  <button
   816→                    onClick={() => togglePinned(t.key)}
   817→                    className={`flex h-7 w-7 items-center justify-center rounded-md shrink-0 transition-colors ${isPinned ? 'text-[var(--j-violet)]' : 'text-[var(--j-text-mute)] hover:text-[var(--j-violet)]'}`}
   818→                    title={isPinned ? 'Unpin' : 'Pin to top'}
   819→                  >
   820→                    <Pin className={`h-3.5 w-3.5 ${isPinned ? 'fill-[var(--j-violet)]' : ''}`} />
   821→                  </button>
   822→                  <div className="flex h-7 w-7 items-center justify-center rounded-md shrink-0" style={{ background: `${t.accent}1a`, border: `1px solid ${t.accent}33`, color: t.accent }}>
   823→                    <Icon className="h-3.5 w-3.5" />
   824→                  </div>
   825→                  <button onClick={() => !isHidden && onNavigate(t.key)} className="flex-1 text-left min-w-0">
   826→                    <div className="text-xs text-[var(--j-text)] truncate">{t.label}</div>

   840→        </div>
   841→        <div className="px-5 py-2.5 border-t border-[var(--j-border)] bg-[var(--j-panel-soft)]/40 flex items-center justify-between">

   850→/* ---------- Global search overlay (Cmd+Shift+F) ---------- */
   851→interface SearchResult {
   852→  id: string;
   853→  type: 'agent' | 'task' | 'memory' | 'comms' | 'skill';
   854→  title: string;
   855→  subtitle: string;
   856→  meta: string;
   857→  color: string;
   858→  href: TabKey;
   859→  score: number;
   860→}
   861→
   862→function GlobalSearch({ open, onClose, onNavigate }: { open: boolean; onClose: () => void; onNavigate: (t: TabKey) => void }) {
   863→  const [q, setQ] = useState('');
   864→  const [results, setResults] = useState<SearchResult[]>([]);
   865→  const [loading, setLoading] = useState(false);
   866→  const [sel, setSel] = useState(0);
   867→  const [savedSearches, setSavedSearches] = useState<string[]>([]);
   868→
   869→  // Load saved searches from localStorage on mount.
   870→  useEffect(() => {
   871→    try {
   872→      const raw = localStorage.getItem('jarvis-saved-searches');
   873→      if (raw) setSavedSearches(JSON.parse(raw));
   874→    } catch { /* ignore */ }
   875→  }, []);
   876→
   877→  const saveSearch = (term: string) => {
   878→    const t = term.trim();
   879→    if (!t) return;
   880→    setSavedSearches((prev) => {
   881→      const next = prev.includes(t) ? prev.filter((s) => s !== t) : [...prev, t].slice(-8);
   882→      try { localStorage.setItem('jarvis-saved-searches', JSON.stringify(next)); } catch { /* ignore */ }
   883→      return next;
   884→    });
   885→  };
   886→
   887→  // Debounced search.
   888→  useEffect(() => {
   889→    if (!open) { setQ(''); setResults([]); setSel(0); return; }
   890→    if (!q.trim()) { setResults([]); return; }
   891→    setLoading(true);
   892→    const id = setTimeout(async () => {
   893→      try {
   894→        const res = await fetch(`/api/search?q=${encodeURIComponent(q)}`, { cache: 'no-store' });
   895→        const json = await res.json();
   896→        setResults(json.results ?? []);
   897→        setSel(0);
   898→      } catch { /* ignore */ } finally {
   899→        setLoading(false);
   900→      }
   901→    }, 200);
   902→    return () => clearTimeout(id);
   903→  }, [q, open]);
   904→
   905→  useEffect(() => {
   906→    if (!open) return;
   907→    const onKey = (e: KeyboardEvent) => {
   908→      if (e.key === 'ArrowDown') { e.preventDefault(); setSel((s) => Math.min(s + 1, results.length - 1)); }
   909→      if (e.key === 'ArrowUp') { e.preventDefault(); setSel((s) => Math.max(s - 1, 0)); }
   910→      if (e.key === 'Enter') { e.preventDefault(); const r = results[sel]; if (r) { onNavigate(r.href); onClose(); } }
   911→    };
   912→    window.addEventListener('keydown', onKey);
   913→    return () => window.removeEventListener('keydown', onKey);
   914→  }, [open, results, sel, onNavigate, onClose]);
   915→
   916→  const typeIcon: Record<string, typeof Bot> = { agent: Bot, task: ListTodo, memory: Database, comms: MessagesSquare, skill: Sparkles };
   917→
   918→  return (
   919→    <AnimatePresence>
   920→      {open && (
   921→        <motion.div className="fixed inset-0 z-50 flex items-start justify-center pt-[10vh] px-4" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }}>
   922→          <div className="absolute inset-0 bg-black/70 backdrop-blur-sm" onClick={onClose} />
   923→          <motion.div initial={{ scale: 0.96, opacity: 0, y: -8 }} animate={{ scale: 1, opacity: 1, y: 0 }} exit={{ scale: 0.96, opacity: 0, y: -8 }} className="relative w-full max-w-2xl jarvis-panel p-0 overflow-hidden">
   924→            <div className="flex items-center gap-2.5 px-4 h-12 border-b border-[var(--j-border)]">
   925→              <Search className="h-4 w-4 text-[var(--j-cyan)]" />
   926→              <input
   927→                autoFocus
   928→                value={q}
   929→                onChange={(e) => setQ(e.target.value)}
   930→                placeholder="Search agents, tasks, memory, comms, skills…"
   931→                className="flex-1 bg-transparent outline-none text-sm text-[var(--j-text)] placeholder:text-[var(--j-text-mute)]"
   932→              />
   933→              {loading && <Loader2 className="h-3.5 w-3.5 animate-spin text-[var(--j-cyan)]" />}
   934→              {q.trim() && (
   935→                <button onClick={() => saveSearch(q)} className="flex items-center gap-1 jarvis-mono text-[9px] uppercase text-[var(--j-cyan)] hover:underline shrink-0">
   936→                  <Star className="h-3 w-3" /> {savedSearches.includes(q.trim()) ? 'saved' : 'save'}
   937→                </button>
   938→              )}
   939→              <kbd className="jarvis-mono text-[9px] px-1.5 py-0.5 rounded border border-[var(--j-border)] text-[var(--j-text-mute)]">ESC</kbd>
   940→            </div>
   941→            <div className="max-h-96 overflow-y-auto jarvis-scroll p-2">
   942→              {results.length > 0 ? (
   943→                results.map((r, i) => {
   944→                  const Icon = typeIcon[r.type] ?? Search;
   945→                  const active = i === sel;
   946→                  return (
   947→                    <button
   948→                      key={r.id}
   949→                      onMouseEnter={() => setSel(i)}
   950→                      onClick={() => { onNavigate(r.href); onClose(); }}
   951→                      className={cn('w-full flex items-center gap-3 px-3 py-2.5 rounded-md text-left transition-colors', active ? 'bg-[var(--j-panel-soft)]' : 'hover:bg-[var(--j-panel-soft)]/60')}
   952→                    >
   953→                      <div className="flex h-8 w-8 items-center justify-center rounded-md shrink-0" style={{ background: `${r.color}1a`, border: `1px solid ${r.color}33`, color: r.color }}>
   954→                        <Icon className="h-4 w-4" />
   955→                      </div>
   956→                      <div className="min-w-0 flex-1">
   957→                        <div className="text-sm text-[var(--j-text)] truncate">{r.title}</div>
   958→                        <div className="text-[11px] text-[var(--j-text-dim)] truncate">{r.subtitle}</div>
   959→                      </div>
   960→                      <div className="text-right shrink-0">
   961→                        <div className="jarvis-mono text-[9px] uppercase px-1.5 py-0.5 rounded" style={{ color: r.color, background: `${r.color}1a` }}>{r.type}</div>
   962→                        <div className="jarvis-mono text-[9px] text-[var(--j-text-mute)] mt-0.5">{r.meta}</div>
   963→                      </div>
   964→                    </button>
   965→                  );
   966→                })
   967→              ) : q.trim() ? (
   968→                <div className="px-3 py-10 text-center">
   969→                  <Search className="h-8 w-8 mx-auto mb-2 text-[var(--j-text-mute)] opacity-40" />
   970→                  <div className="text-sm text-[var(--j-text-mute)]">No results for "{q}"</div>
   971→                </div>
   972→              ) : savedSearches.length > 0 ? (
   973→                <div className="px-3 py-4">
   974→                  <div className="jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)] mb-2 flex items-center gap-1.5"><Star className="h-3 w-3" /> saved searches</div>
   975→                  <div className="flex flex-wrap gap-2">
   976→                    {savedSearches.map((s) => (
   977→                      <button key={s} onClick={() => setQ(s)} className="group flex items-center gap-1.5 text-xs px-2.5 py-1.5 rounded-md border border-[var(--j-border)] bg-[var(--j-panel-soft)] text-[var(--j-text-dim)] hover:border-[var(--j-cyan)] hover:text-[var(--j-cyan)] transition-colors">
   978→                        <Search className="h-3 w-3" /> {s}
   979→                        <span onClick={(e) => { e.stopPropagation(); saveSearch(s); }} className="text-[var(--j-text-mute)] hover:text-[var(--j-red)] ml-0.5"><X className="h-2.5 w-2.5" /></span>
   980→                      </button>
   981→                    ))}
   982→                  </div>
   983→                </div>
   984→              ) : (
   985→                <div className="px-3 py-10 text-center">
   986→                  <div className="text-sm text-[var(--j-text-mute)]">Type to search across the fleet</div>
   987→                  <div className="jarvis-mono text-[10px] text-[var(--j-text-mute)] mt-1">agents · tasks · memory · comms · skills</div>
   988→                </div>
   989→              )}
   990→            </div>
   991→            <div className="px-4 py-2 border-t border-[var(--j-border)] bg-[var(--j-panel-soft)]/40 flex items-center justify-between jarvis-mono text-[9px] uppercase text-[var(--j-text-mute)]">
   992→              <span>{results.length} results</span>
   993→              <span className="flex items-center gap-2">
   994→                <span>↑↓ navigate</span>
   995→                <span>⏎ open</span>
   996→              </span>
   997→            </div>
   998→          </motion.div>
   999→        </motion.div>
  1000→      )}
  1001→    </AnimatePresence>
  1002→  );
  1003→}
